万物之父

关键字： object
概念： object是所有类型的基类，它是一个类（引用类型）
作用：
1.可以利用里式替换原则，用object容器装所有对象
2.可以用来表示不确定类型，作为函数参数类型

class Father
{

}

class Son : Father
{
    public void Speak()
    {

    }
}

internal class Program
{
    static void Main(string[] args)
    {
        Father f = new Son();
        if( f is Son )
        {
            ( f as Son ).Speak();
        }

        //引用类型
        object o = new Son();
        if( o is Son )
        {
            (o as Son ).Speak();
        }
        //值类型
        object o2 = 1;
        //用强转进行数值使用
        int i = (int)o2;

        //特殊的string类型
        object o3 = "123123";
        string s = o3 as string; //o3.ToString();

        //数组类型
        object o4 = new int[10];
        int[] arr = o4 as int[]; //(int[])o4;

    }
}

拆箱装箱

发生条件：
装箱： object中存入值类型
拆箱： object转为值类型

装箱：
把值类型用引用类型存储
栈内存会迁移到堆内存中

拆箱：
把引用类型存储的值类型取出来
堆内存会迁移到栈内存中

好处：不确定类型时可以方便参数的存储和传递
坏处：存在内存迁移，增加性能消耗

internal class Program
{
    static void Main(string[] args)
    {
        //装箱
        object v = 3;
        //拆箱
        int intValue = (int)v;

        //传入object类数组
        TestFun(1, 1.5f , 2, "123", new int[10]);
    }

    static void TestFun( params object[] array )
    {

    }
}
