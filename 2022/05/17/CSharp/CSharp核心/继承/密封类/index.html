密封类

密封类

基本概念

密封类：使用sealed密封关键字修饰的类
作用：让类无法再被继承

sealed class 类名

作用

不允许最底层子类被继承，保证程序的规范性、结构性、安全性

练习题

没设置自定义乘客数量、途径站点数

using System;

namespace Lesson16_密封类
{

    sealed class Carrier
    {
        //两站间隔1km
        int s = 1000;
        //载具速度
        int speed;
        //最大速度100m/min
        int speedMax = 100;
        //最大可乘10人,包含司机
        int maxNumber = 10;
        
        //上车人数
        public int onNum = 0;
        //下车人数
        public int offNum = 0;
        //乘客数组
        public bool[] passengers = new bool[10];

        //上下车人数的随机数,载具速度的随机数
        Random r = new Random();

        //逻辑上先上车再下车
        //上车 + 下车
        public void GetOnOff(ref int nowNum)
        { 
            //上车人数不能超过总人数
            onNum = r.Next(10 - nowNum);
            Console.Write("上车{0}人  ", onNum);
            for (int i = nowNum; i < nowNum + onNum; i++)
            {
                passengers[i] = true;
            }
            //下车人数不能超过原已上车人数
            offNum = r.Next(nowNum); 
            Console.Write("下车{0}人  ", offNum);
            for (int i = nowNum + onNum - 1; i >= nowNum + onNum - offNum; i--)
            {
                passengers[i] = false;
            }
            //更新已上车人数
            nowNum = nowNum + onNum - offNum;
        }

        //行驶 打印多久到下一站 按任意键到下一站
        public void Drive()
        {
            //载具速度
            speed = r.Next(50, speedMax + 1);
            Console.WriteLine("耗时{0}分钟,到达下一站", s/speed);
        
        }

        //车祸
        public void Accident()
        {
            Console.WriteLine("很不幸,发生了车祸");
            Environment.Exit(0);
        }
    }


    internal class Program
    {
        static void Main(string[] args)
        {
            //定义一个随机数,判断是否车祸
            Random r = new Random();
            //默认一开始没人上车,即乘客全为false
            Carrier c = new Carrier();
            //已上车人数 初始为0
            int nowNum = 0;
            //计数到站数,设一共10站
            int set = 0;
            for (int i = 0; i < c.passengers.Length; i++)
            {
                c.passengers[i] = false;
            }
            Console.WriteLine("现在从始发站开始出发,车上无乘客");

            while(true)
            {
                Console.WriteLine("请按任意键前往下一站");
                Console.WriteLine("");
                Console.ReadKey(true);
                
                //95%的概率正常行驶,5%的概率出车祸
                if ( r.Next(101) <= 95 )
                {
                    c.Drive();
                    c.GetOnOff(ref nowNum);
                    Console.WriteLine("现在车上有{0}人", nowNum);
                    set++;
                }
                else
                {
                    c.Accident();
                }
                if( set == 10 )
                {
                    Console.WriteLine("\n已到达终点站");
                    Environment.Exit(0);
                }
            } 
        }
    }
}
