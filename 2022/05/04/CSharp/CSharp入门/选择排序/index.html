选择排序

重点知识点：
选择排序
1.基本概念： 新建中间商 遍历依次比较找出极值 放入目标位置
2.套路写法：两层循环（外层轮数 内层寻找 初始索引 记录极值）
3.优化： 已经完成比较的不参与比较 加一个bool标识判断是否已经有序

练习题

以下是答案：

using System;

namespace Lesson14_选择排序练习题
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("选择排序练习题");

            #region 练习题1   

            //生成随机数组
            Random r = new Random();
            int[] arr = new int[20];
            for (int i = 0; i < arr.Length; i++)
            {
                arr[i] = r.Next(0, 101);
            }

            //选择排序升序排序
            Console.WriteLine("选择排序升序排序结果:");
            for (int m = 0; m < arr.Length; m++)
            {
                int index = 0;
                int temp;
                for (int n = 0; n < arr.Length - m; n++)
                {
                    if (arr[index] < arr[n])
                    {
                        index = n;
                    }
                }
                if(index != arr.Length - m -1 )
                {
                    temp = arr[index];
                    arr[index] = arr[arr.Length - m - 1];
                    arr[arr.Length - m - 1] = temp;
                }
            }
            for (int i = 0; i < arr.Length; i++)
            {
                Console.Write(arr[i] + " ");
            }

            //中间间隔
            Console.WriteLine();
            Console.WriteLine("*************");

            //冒泡排序降序排序
            Console.WriteLine("选择排序降序排序结果:");
            for (int m = 0; m < arr.Length; m++)
            {
                int index = 0;
                int temp;
                for (int n = 0; n < arr.Length - m; n++)
                {
                    if (arr[index] > arr[n])
                    {
                        index = n;
                    }
                }
                if (index != arr.Length - m - 1)
                {
                    temp = arr[index];
                    arr[index] = arr[arr.Length - m - 1];
                    arr[arr.Length - m - 1] = temp;
                }
            }
            for (int i = 0; i < arr.Length; i++)
            {
                Console.Write(arr[i] + " ");
            }

            #endregion
        }
    }
}
