冒泡排序

重点知识点：
冒泡排序
1.基本概念： “无序”数组中两两相邻的数不断比较与交换直到“有序”
2.套路写法：两层循环（外层轮数 内层比较 满足则交换）
3.优化： 已经完成比较的不参与比较 加一个bool标识判断是否已经有序

练习题

以下是答案：

using System;

namespace Lesson13_冒泡排序练习题
{
    internal class Program
    {

        #region 练习题2
        //冒泡排序函数
        static void PaiXu( int[] arr, bool upDown)
        {
            //小优化 在循环外声明变量 可以提高一些性能
            bool isSort;
            int temp;
            for (int m = 0; m < arr.Length; m++)
            {
                isSort = false;

                for (int n = 0; n < arr.Length - 1 - m; n++)
                {
                    if(upDown)
                    {
                        if (arr[n] > arr[n + 1])
                        {
                            isSort = true;
                            temp = arr[n];
                            arr[n] = arr[n + 1];
                            arr[n + 1] = temp;
                        }
                    }
                    else
                    {
                        if (arr[n] < arr[n + 1])
                        {
                            isSort = true;
                            temp = arr[n];
                            arr[n] = arr[n + 1];
                            arr[n + 1] = temp;
                        }
                    }
                }
                if (!isSort)
                {
                    break;
                }
            }
            if (upDown)
            {
                Console.WriteLine("冒泡排序升序排序结果:");
            }
            else
            {
                Console.WriteLine("冒泡排序降序排序结果:");
            }
            for (int i = 0; i < arr.Length; i++)
            {            
                Console.Write(arr[i] + " ");
            }
        }


        #endregion

        static void Main(string[] args)
        {
            Console.WriteLine("冒泡排序练习题");

            #region 练习题1

            //判断是否已经有序 标识符
            bool isSort = false;

            //生成随机数组
            Random r = new Random();
            int[] arr1 = new int[20];
            for (int i = 0; i < arr1.Length; i++)
            {
                arr1[i] = r.Next(0, 101);
            }

            //冒泡排序升序排序
            Console.WriteLine("冒泡排序升序排序结果:");
            for(int m = 0; m < arr1.Length; m++)
            {
                isSort = false;
                for(int n = 0; n < arr1.Length - 1 - m; n++)
                {
                    if(arr1[n] > arr1[n + 1])
                    {
                        isSort = true;
                        int temp = arr1[n];
                        arr1[n] = arr1[n + 1];
                        arr1[n + 1] = temp;
                    }
                }
                if( !isSort )
                {
                    break;
                }
            }
            for(int i = 0; i < arr1.Length; i++)
            {
                Console.Write(arr1[i] + " ");
            }

            //中间间隔
            Console.WriteLine();
            Console.WriteLine("*************");

            //冒泡排序降序排序
            Console.WriteLine("冒泡排序降序排序结果:");
            for (int m = 0; m < arr1.Length; m++)
            {
                isSort = false;
                for (int n = 0; n < arr1.Length - 1 - m; n++)
                {
                    if (arr1[n] < arr1[n + 1])
                    {
                        isSort = true;
                        int temp = arr1[n];
                        arr1[n] = arr1[n + 1];
                        arr1[n + 1] = temp;
                    }
                }
                if (!isSort)
                {
                    break;
                }
            }
            for (int i = 0; i < arr1.Length; i++)
            {
                Console.Write(arr1[i] + " ");
            }

            #endregion

            //中间间隔
            Console.WriteLine();
            Console.WriteLine("*************");

            #region 练习题2

            //生成数组
            int[] arr2 = new int[20];
            for (int i = 0; i < arr2.Length; i++)
            {
                arr2[i] = r.Next(0, 101);
            }

            //true代表升序排序 false代表降序排序
            PaiXu( arr2, true);
            //中间间隔
            Console.WriteLine();
            Console.WriteLine("*************");
            PaiXu( arr2, false);

            #endregion

        }
    }
}
