    /// <summary>

    struct Vector2

        public int y;

        public Vector2(int x, int y)
        {
            this.x = x; this.y = y;

    //格子结构体
    struct Grid
    {
        public E_GridType gridType; //格子类型
        public Vector2 position; //格子位置
        //构造函数  输入格子位置  格子初始化
        public Grid(E_GridType gridType, int x, int y)
        {
            this.gridType = gridType;
            position.x = x; 
            position.y = y;
        }
        //画格子
        public void Draw()
        {
            //控制光标位置  打印选项
            Console.SetCursorPosition(position.x, position.y);
            switch (gridType)
            {
                //普通格子
                case E_GridType.Normal:
                    Console.ForegroundColor = ConsoleColor.White;
                    Console.Write('□');
                    break;
                //炸弹
                case E_GridType.Bomb:
                    Console.ForegroundColor = ConsoleColor.Blue;
                    Console.Write("‖");
                    break;
                //暂停
                case E_GridType.Pause:
                    Console.ForegroundColor = ConsoleColor.Red;
                    Console.Write("●");
                    break;
                //时空隧道
                case E_GridType.Tunnel:
                    Console.ForegroundColor = ConsoleColor.White;
                    Console.Write("¤");
                    break;
            }
        }
    }

    struct Map
    {
        //用一个格子结构体数组 表示地图里的所有格子
        public Grid[] grids;

        //地图初始化构造函数
        //x, y为棋盘起始点
        public Map(int x, int y)
        {
            grids = new Grid[84];

            Random r = new Random();
            int randomNum;
            for (int i = 0; i < 84; i++)
            {
                //应该初始化 格子类型
                randomNum = r.Next(0, 100);

                //设置类型 普通格子
                //有85%几率 是普通格子 (首尾两个格子 必为普通格子)
                if(randomNum < 85 || i == 0 || i == 95)
                {
                    grids[i].gridType = E_GridType.Normal;
                }
                //有5%几率 是炸弹
                if (randomNum >= 85 && randomNum < 90)
                {
                    grids[i].gridType = E_GridType.Bomb;
                }
                //有5%几率 是暂停
                if (randomNum >= 90 && randomNum < 95)
                {
                    grids[i].gridType = E_GridType.Pause;
                }
                //有5%几率 是时空隧道
                if (randomNum >= 95 && randomNum < 100)
                {
                    grids[i].gridType = E_GridType.Tunnel;
                }

                //设置位置
                grids[i].position = new Vector2(x, y);
                //每次循环都应该一定规则去变化位置(每次少4个步长)
                if( i <= 12 || (i > 56 && i <= 64)  )
                {
                    y++;
                }
                else if( (i > 12 && i <= 30) || (i > 64 && i <= 74) )
                {
                    x += 2;
                }
                else if( (i > 30 && i <= 42) || (i > 74 && i <= 78) )
                {
                    y--;
                }
                else if(( i > 42 && i <= 56) || (i > 78 && i <= 84) )
                {
                    x -= 2;
                }
            }
        }

        public void Draw()
        {
            for (int i = 0; i < grids.Length; i++)
            {
                grids[i].Draw();
            }
        }
    }

        static void DrawPlayer(Player you, Player PC, Map map)
        {
            //重合的话 Player结构体外部处理
            if(you.nowIndex == PC.nowIndex)
            {
                //得到重合的位置
                Grid grid = map.grids[you.nowIndex];
                Console.SetCursorPosition(grid.position.x, grid.position.y);
                Console.ForegroundColor = ConsoleColor.Green;
                Console.WriteLine("◎");
            }
            //不重合的话 Player结构体内部处理 
            else
            {
                you.DrawMove(map);
                PC.DrawMove(map);
            }
        }

            Console.SetWindowSize(w, h);
            //设置缓冲区大小
            Console.SetBufferSize(w, h);
        }

        #endregion

        #region 红色墙壁
        static void DrawWall(int w, int h)
        {
            //设置颜色为红色
            Console.ForegroundColor = ConsoleColor.Red;
            //画墙
            //上方墙 下方墙 中间墙1 中间墙2
            for (int i = 0; i <= w - 2; i += 2)
            {
                Console.SetCursorPosition(i, 0); //上方墙
                Console.Write('■');
                Console.SetCursorPosition(i, h - 1); //下方墙
                Console.Write('■');
                Console.SetCursorPosition(i, h - 11); //中间墙1
                Console.Write('■');
                Console.SetCursorPosition(i, h - 6); //中间墙2
                Console.Write('■');
            }
            //左边墙 右边墙
            for (int i = 0; i <= h - 1; i++)
            {
                Console.SetCursorPosition(0, i); //左边墙
                Console.Write('■');
                Console.SetCursorPosition(w - 2, i); //右边墙
                Console.Write('■');
            }
        }
        #endregion


    }
