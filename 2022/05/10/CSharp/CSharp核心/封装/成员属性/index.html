成员属性

成员属性

基本概念

1.用于保护成员变量
2.为成员属性的获取和赋值添加逻辑处理
3.解决3P的局限性
public - 内外访问
private - 内部访问
protected - 内部和子类访问
属性可以让成员变量在外部 只能获取不能修改 或 只能修改不能获取

基本语法

class Person
{
    private string name;

    //属性的命名一般使用 帕斯卡命名法
    //访问修饰符 属性类型 属性名
    public string Name
    {
        get
        {
            //可以在返回前添加一下逻辑规则（解密）
            //get意味着这个属性可以获取的内容 需要return内容
            return name;
        }
        set
        {
            //可以在设置前添加一下逻辑规则（加密）
            // value 关键字 用于表示外部传入的值
            name = value;                 
        }
    }
}

使用

Person p = new Person();
p.Name = "季宝";

get和set前可以加访问修饰符

解决3P局限性
1.默认不加 会使用属性声明时的访问权限
2.加的访问修饰符要低于属性的访问权限（public的访问权限是最高级的）
3.不能让get和set的访问权限都低于属性的权限（逻辑上外部的访问修饰符无效，报错）

get和set只能有一个

只有一个时 没有必要在前面加访问修饰符
且一般只有get 没有set

自动属性

属性语句块中只有get和set
作用：
外部能得不能改的特征
如果类中有一个特征是只希望外部能得不能改的 又没有什么特别处理
那么可以直接使用自动属性

class Person
{
    private Height;
    //不用专门去声明成员变量 直接通过属性来表示特殊的成员变量
    public float Height
    {
        //没有在get和set中写逻辑的需求或者想法
        get;
        set;
    }

    //等价于
    public float Height { get; set; }

}

练习题

using System;

namespace Lesson5_成员属性
{
    class Student
    {
        private string name;
        private string sex;
        private int age;
        private int csharpScore;
        private int unityScore;

        public string Name { get; set; }
        public string Sex 
        { 
            get
            {
                if (sex != "男" && sex != "女")
                {
                    Console.WriteLine("请输入正确的性别,默认男");
                    return "男";
                }
                else
                {
                    return sex;
                }
                
            }
            set
            {
                sex = value;
            }
        }
        public int Age 
        { 
            get
            {
                return age;    
            }
            set
            {
                if (value >= 0 && value <= 150)
                {
                    age = value;
                }
                else
                {
                    Console.WriteLine("请输入0~150之间的年龄,默认20");
                    age = 20;
                }   
            }
        }
        public int CSharpScore
        {
            get
            {
                if (csharpScore >= 0 && csharpScore <= 150)
                {
                    return csharpScore;
                }
                else
                {
                    Console.WriteLine("请输入0~100之间的成绩,默认60");
                    return 60;
                }
            }
            set
            {
                csharpScore = value;
            }
        }
        public int UnityScore
        {
            get
            {
                if (unityScore >= 0 && unityScore <= 150)
                {
                    return unityScore;
                }
                else
                {
                    Console.WriteLine("请输入0~100之间的成绩,默认60");
                    return 60;
                }
            }
            set
            {
                unityScore = value;
            }
        }

        public void Hello1()
        {
            Console.WriteLine("我叫{0},今年{1}岁了", Name, Age);
        }
        public void Hello2()
        {
            Console.WriteLine("我是{0}生,我的总分是{1},平均分是{2}", Sex, CSharpScore + UnityScore, (CSharpScore + UnityScore)/2);
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Student s1 = new Student();
            s1.Name = "季宝";
            s1.Sex = "男";
            s1.Age = 170;
            s1.CSharpScore = 80;
            s1.UnityScore = 80;
            s1.Hello1();
            s1.Hello2();
            Student s2 = new Student();
        }
    }
}
