多态的概念

多态： 字面意思“多种状态”
让继=继承同一父类的子类们，在执行相同方法时有不同表现（状态）

主要目的：
同一父类的对象 执行相同行为（方法）有不同的表现

解决的问题：
让同一个对象有唯一行为的特征

解决的问题

使同一种方法在不同对象下有不同表现

class Father
{
    public void SpeakName()
    {
        Console.WriteLine("Father的方法");
    }
}

class Son : Father
{
    public new void SpeakName()
    {
        Console.WriteLine("Son的方法");
    }
}

internal class Program
{
    static void Main(string[] args)
    {
        Father f = new Son();
        f.SpeakName(); //打印Father的方法
        (f as Son).SpeakName(); //打印Son的方法
    }
}

多态的实现

函数重载 就是一种编译时的多态

运行时的多态：
(VOB, 抽象函数, 接口)
V: virtual 虚函数
O: override 重写
B: base 父类

class GameObject
{
    public string name;
    public GameObject(string name)
    {
        this.name = name;
    }

    //虚函数 可以被子类重写
    public virtual void Atk()
    {
        Console.WriteLine("游戏对象进行攻击");
    }
}

class Player : GameObject
{
    public Player(string name): base (name)
    {

    }
    public override void Atk()
    {
        //base代表父类,可以通过base来保留父类的行为
        base.Atk();
        Console.WriteLine("玩家对象进行攻击");
    }
}

internal class Program
{
    static void Main(string[] args)
    {
        GameObject p = new Player("大逼");
        p.Atk(); //先打印父类行为"游戏对象进行攻击",再打印子类行为"玩家对象进行攻击"
    }
}

using System;

namespace Lesson17_VOB
{
    //鸭子
    class Duck
    {
        public virtual void Quack()
        {
            Console.WriteLine("真的鸭子嘎嘎叫");
        }
    }
    
    //木头鸭子
    class WoodDuck : Duck
    {
        public override void Quack()
        {
            Console.WriteLine("木头鸭子吱吱叫");
        }
    }

    //木头鸭子
    class RubberDuck : Duck
    {
        public override void Quack()
        {
            Console.WriteLine("橡皮鸭子唧唧叫");
        }
    }


    internal class Program
    {
        static void Main(string[] args)
        {
            Duck d = new Duck();
            d.Quack();
            Duck wd = new WoodDuck();
            wd.Quack();
            Duck rd = new RubberDuck();
            rd.Quack();
        }
    }
}

using System;

namespace Lesson17_VOB
{
    //图形
    class Figure
    {
        //求面积
        public virtual void Area()
        {
            
        }
        //求周长
        public virtual void Perimeter()
        {
            
        }
    }
    
    //矩形
    class Rectangular : Figure
    {
        public int l;
        public int w;
        public Rectangular(int l, int w)
        {
            this.l = l;
            this.w = w;
        }
        public override void Area()
        {
            int s = l * w;
            Console.WriteLine("该矩形面积为{0}", s);
        }
        public override void Perimeter()
        {
            int c = 2 * ( l + w );
            Console.WriteLine("该矩形周长为{0}", c);
        }
    }

    //矩形
    class Square : Figure
    {
        public int a;
        public Square(int a)
        {
            this.a = a;
        }
        public override void Area()
        {
            int s = a * a;
            Console.WriteLine("该矩形面积为{0}", s);
        }
        public override void Perimeter()
        {
            int c = 4 * a;
            Console.WriteLine("该矩形周长为{0}", c);
        }
    }

    //圆形
    class Circular : Figure
    {
        public int r;
        const float PI = 3.1415926f;
        public Circular(int r)
        {
            this.r = r;
        }
        public override void Area()
        {
            float s = PI * r * r;
            Console.WriteLine("该矩形面积为{0}", s);
        }
        public override void Perimeter()
        {
            float c = 2 * r * PI;
            Console.WriteLine("该矩形周长为{0}", c);
        }
    }


    internal class Program
    {
        static void Main(string[] args)
        {
            Figure r = new Rectangular(2, 4);
            r.Area();
            r.Perimeter();
            Figure s = new Square(3);
            s.Area();
            s.Perimeter(); 
            Figure c = new Circular(5);
            c.Area();
            c.Perimeter();
        }
    }
}
