泛型约束

泛型约束

概念

让泛型的类型有一定的限制
关键字：where
泛型约束共有6种
1.值类型 where 泛型字母:struct
2.引用类型 where 泛型字母:class
3.存在无参公共构造函数 where 泛型字母:new()
4.某个类本身或者其派生类 where 泛型字母:类名
5.某个接口的派生类型 where 泛型字母:接口名
6.另一个泛型类型本身或者派生类型 where 泛型字母:另一个泛型字母

where 泛型字母:(约束的类型)

各个泛型约束

1.值类型约束

class Test1<T> where T:struct
{
    public T value;
    public void TestFun<K>(K k) where K:struct
    {

    }
}

//int 和 float 都是struct类型（值类型，初值不能为空）
Test1<int> t1 = new Test1<int>();
t1.TestFun<float>(1.3f);

2.引用类型约束

class Test2<T> where T:class
{
    public T value;
    public void TestFun<K>(K k) where K:class
    {

    }
}

//Random 和 object 都是class类型
Test2<Random> t2 = new Test2<Random>();
t2.TestFun<object>(new object());

3.存在无参公共构造函数约束（非抽象类）

class Test3<T> where T:new()
{
    public T value;
    public void TestFun<K>(K k) where K:new()
    {

    }
}

class Test1
{

}

class Test2
{
    public Test2(int a)
    {

    }
}

//Test1可以（有公共无参构造函数） Test2不行（公共有参、私有无参、抽象类都不行）
Test3<Test1> t3 = new Test3<Test1>();

4.某个类本身或者其派生类 约束

class Test4<T> where T:Test1
{
    public T value;
    public void TestFun<K>(K k) where K:Test1
    {

    }
}

class Test3 : Test1
{

}

//Test1可以（类） Test3可以（派生类） Test2不行
Test4<Test1> t4 = new Test4<Test1>();
Test4<Test3> t4 = new Test4<Test3>();

5.某个接口的派生类型约束

interface IFly
{

}

class Test4 : IFly
{

}
class Test5<T> where T:IFly
{
    public T value;
    public void TestFun<K>(K k) where K:IFly
    {

    }
}

//IFly可以（接口） Test4可以（派生类） 
Test5<IFly> t5 = new Test5<IFly>();
t5.value = new Test4(); //里式替换
Test4<Test4> t5 = new Test4<Test4>();

6.另一个泛型类型本身或者派生类型约束

//T要么是U,要么是U的派生类型
class Test6<T,U> where T:U
{
    public T value;
    public void TestFun<K,V>(K k) where K:V
    {

    }
}

//Test4是IFly的派生类型 
Test6<Test4, IFly> t6 = new Test6<Test4, IFly>();

        private static T instance = new T();

        {
            get { return instance; }

        public int value = 10;

    {
        private static Test instance = new Test();

        {

        }

        public static Test Instance
        {
            get
            {
                return instance;
            }
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            GameMgr.Instance.value = 10;
        }
    }
}

题2

using System;

namespace Lesson6_泛型约束
{
    class ArrayList<T>
    {
        private T[] array;
        private int count;

        /// <summary>
        /// 获取已存数量
        /// </summary>
        public int Count
        {
            get { return count; }
        }

        /// <summary>
        /// 获取容量
        /// </summary>
        public int Capacity { get { return array.Length; } }

        public ArrayList()
        {
            count = 0;
            //一开始的容量为16
            array = new T[16];
        }

        //增
        public void Add(T value)
        {
            //判断是否需要扩容
            if(count >= Capacity)
            {
                T[] newArray = new T[Capacity << 1];
                for (int i = 0; i < Capacity; i++)
                {
                    newArray[i] = array[i];
                }
                array = newArray;
            }
            //不需要扩容
            array[count++] = value;
        }

        //删
        public void Remove(T value)
        {
            int index = -1;
            for (int i = 0; i < Count; i++)
            {
                //因为不是所有的类型都重载了==,所以无法用==判断
                if(value.Equals(array[i]))
                {
                    index = i;
                    break;
                }
            }
            if(index != -1)
            {
                for (int i = index; i < Count; i++)
                {
                    array[i] = array[i + 1];
                }
                array[Count] = default(T);
                --count;
            }
        }

        //查
        public void RemoveAt(int index)
        {
            if(index < 0 || index >= Count)
            {
                return;
            }
            for (int i = index; i < Count; i++)
            {
                array[i] = array[i + 1];
            }
            array[Count] = default(T);
            --count;
        }

        //改
        public T this[int index]
        {
            get 
            {
                if (index < 0 || index >= Count)
                {
                    return default(T);
                }
                return array[index]; 
            }
            set 
            {
                if (index < 0 || index >= Count)
                {
                    return;
                }
                array[index] = value; 
            }
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            ArrayList<int> array = new ArrayList<int>();
            Console.WriteLine("{0},{1}", array.Count, array.Capacity);
            array.Add(1);
            array.Add(2);
            array.Add(3);
            Console.WriteLine("{0},{1}", array.Count, array.Capacity);

            Console.WriteLine(array[1]);
            Console.WriteLine(array[-1]);

            array.RemoveAt(0);
            Console.WriteLine("{0},{1}", array.Count, array.Capacity);
            for (int i = 0; i < array.Count; i++)
            {
                Console.WriteLine(array[i]);
            }

            Console.WriteLine(array[0]);
            array[0] = 5;
            Console.WriteLine(array[0]);
        }
    }
}
