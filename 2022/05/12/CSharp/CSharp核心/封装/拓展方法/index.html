拓展方法

拓展方法

基本概念：
为现有非静态的变量类型 添加新方法

作用：
1.提升程序拓展性
2.不需要在对象中重新写方法
3.不需要继承来添加方法
4.为别人封装的类型写额外的方法

特点：
1.一定是写在静态类中
2.一定是一个静态函数
3.第一个参数为拓展目标
4.第一个参数用this修饰

基本语法

//声明
static class Tools
{
    //访问修饰符 static 返回值 函数名(this 拓展类名 参数名, 参数类型 参数名, 参数类型 参数名, ......)
    public static void SpeakValue(this int value)
    //为int拓展了一个成员方法
    //成员方法 是需要实例化对象后 才能使用的
    //value 代表使用该方法的 实例化对象
    {
        //拓展的方法的逻辑
        Console.WriteLine("为int拓展的方法" + value);
    }

    public static void SpeakStringInfo(this string str, string str2, string str3)
    //为string拓展了一个成员方法
    //str 代表使用该方法的 实例化对象
    //str2, str3 代表传进去的参数
    {
        Console.WriteLine("为string拓展的方法");
        Console.WriteLine("调用方法的对象" + str);
        Console.WriteLine("传的参数" + str2 + str3);
    }
}

//使用
internal class Program
{
    static void Main(string[] args)
    {
        //i就是int这一拓展类 实例化的对象
        int i = 10;
        i.SpeakValue();

        //str就是string这一拓展类 实例化的对象
        string str = "000";
        str.SpeakStringInfo("111", "222");
    }
}

为自定义的类型拓展方法

//声明
static class Tools
{
    public static void Fun3(this Test t)
    //为Test类的t拓展了一个成员方法
    {
        Console.WriteLine("为test拓展的方法");
    }
}

class Test
{
    public int i = 10;
    public void Fun1()
    {
        Console.WriteLine("123");
    }
    public void Fun2()
    {
        Console.WriteLine("456");
    }
}

//使用
internal class Program
{
    static void Main(string[] args)
    {
        Test t = new Test();
        t.Fun3();
    }
}

注意：
拓展方法名若和原有方法名相同，会默认使用原有方法名

练习题

题1

using System;

namespace Lesson9_拓展方法
{
    //为整形拓展一个求平方的方法
    static class Tools
    {
        //平方返回值int
        public static int Square(this int value)
        {
            return value * value;
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            Console.Write("请输入一个整数：");
            int i = int.Parse(Console.ReadLine());
            Console.WriteLine("它的平方是{0}", i.Square());
        }
    }
}

题2

using System;

namespace Lesson9_拓展方法
{
    //声明一个玩家类
    //因为没写敌人，逻辑写的简单点
    class Player
    {
        public string Name { get; set; }
        public int Hp { get; set; }
        public int Atk { get; set; }
        public int Def { get; set; }

        //攻击方法
        public void Attack()
        {
            //攻击能提升气势，增加攻击力
            Atk += 1;
        }
        //移动方法
        public void Move()
        {
            //移动就会找到宝箱打开，要么获得武器，要么获得防具
            Random r = new Random();
            //一半概率为武器
            if(r.Next(1, 101) <= 50)
            {
                Atk += 50;
            }
            //一半概率为防具
            else
            {
                Def += 50;
            }
        }
        //受伤方法
        public void Injured()
        {
            //受伤掉5血
            Hp -= 5;
        }
    }

    //为玩家拓展一个自杀的方法
    static class Tools
    {
        public static void Suicide(this Player p)
        {
            p.Hp = 0;
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            Player p = new Player();
            p.Name = "大逼";
            p.Hp = 100;
            p.Atk = 10;
            p.Def = 10;

            p.Attack();
            p.Move();
            p.Injured();
            if(p.Hp > 0)
            {
                Console.WriteLine("{0}现在的状态为血量{1},攻击力{2},防御力{3}", p.Name, p.Hp, p.Atk, p.Def);
            }
            else if (p.Hp <= 0)
            {
                Console.WriteLine("{0}已死亡", p.Name);
            }
            p.Suicide();
            if (p.Hp > 0)
            {
                Console.WriteLine("{0}现在的状态为血量{1},攻击力{2},防御力{3}", p.Name, p.Hp, p.Atk, p.Def);
            }
            else if (p.Hp <= 0)
            {
                Console.WriteLine("{0}已死亡", p.Name);
            }
        }
    }
}
