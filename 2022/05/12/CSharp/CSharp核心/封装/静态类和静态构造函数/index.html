静态类和静态构造函数

静态类

基本概念：
用static修饰的类

特点：
1.只能包含静态成员
2.不能被实例化

作用：
1.将常用的静态成员写在静态类中 方便使用
2.静态类不能被实例化，更能体现工具性、唯一性
Console就是一个静态类

static class TestStatic
{
    //静态成员变量
    public static int testIndex = 0;

    public static void TestFun()
    {

    } 

    public static int TestIndex()
    {
        get;
        set;
    }

静态构造函数

基本概念：
在构造函数前加上 static 修饰

特点：
1.静态类和普通类都可以有
2.不能使用访问修饰符
3.不能有参数
4.只会自动调用一次

作用：
在静态构造函数中初始化 静态变量

//静态类中的静态构造函数
static class StaticClass
{
    public static int testInt = 100;
    public static int testInt2 = 100;

    static StaticClass()
    {
        //只会在初次使用的时候 调用一次
        Console.WriteLine("静态构造函数");
        testInt = 200;
        testInt2 = 200;
    } 
}

//普通类中的静态构造函数
class Test
{
    public static int testInt = 200;
    static Test()
    {
        Console.WriteLine("静态构造");
    }
    //这不是重载 这是两个不同的函数
    public Test()
    {
        Console.WriteLine("普通构造");
    }
}

练习题

using System;

namespace Lesson8_静态类和静态构造函数
{
    //数学计算静态类
    static class Calc
    {
        public static float PI =3.145926f;



        //计算圆的面积
        public static float CircleS(int r)
        {
            return PI * r * r;
        }

        //计算圆的周长
        public static float CircleL(int r)
        {
            return PI * r * 2;
        }

        //计算矩形的面积
        public static float RectangleS(int l, int w)
        {
            return l * w;
        }

        //计算矩形的周长
        public static float RectangleL(int l, int w)
        {
            return 2 * (l + w);
        }

    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("半径为3的圆面积是{0}", Calc.CircleS(3));
            Console.WriteLine("半径为3的圆周长是{0}", Calc.CircleL(3));
            Console.WriteLine("长为3宽为3的矩形面积是{0}", Calc.RectangleS(3, 3));
            Console.WriteLine("长为3宽为3的矩形周长是{0}", Calc.RectangleL(3, 3));
        }
    }
}
