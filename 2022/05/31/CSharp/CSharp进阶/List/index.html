List

List

概念

List是一个C#为我们封装好的类，
它的本质是一个可变类型的泛型数组。

声明

using System.Collections.Generic;
    List<类型> list = new List<类型>();

增

//增加一个元素
list.Add(类型);
//批量增加，把另一个list容器中的元素加到后面
list.AddRange(另一个list);

删

//移出指定元素，从头找，找到删
array.Remove(元素);
//移除指定位置元素
array.RemoveAt(索引值);
//清空数组
array.Clear();

查

//得到指定位置的元素
Console.WriteLine(list[0]);
//查询元素是否存在
array.Contains(元素); //存在为真,不存在为假
//正向查找元素位置,正向第一个找到的位置
array.IndexOf(元素); //返回索引值,没找到返回-1 
//正向查找元素位置,反向第一个找到的位置
array.LastIndexOf(元素); //返回索引值,没找到返回-1

改

//直接改指定位置的元素
array[索引值] = 元素;

遍历

//数组长度
list.Count
//数组容量
list.Capacity
//遍历
for( i = 0; i < list.Count; i++ )
{
    //需要的逻辑
}
//迭代器遍历 把array里的元素放到item里
foreach (object item in list)
{
    //需要的逻辑 
}

题3

子类默认先调用父类的无参构造函数

using System;
using System.Collections.Generic;

namespace Lesson7_List
{
    abstract class Monster
    {
        //静态列表 不会改变
        public static List<Monster> monsters = new List<Monster>();
        public Monster()
        {
            monsters.Add(this);
        }
        abstract public void Atk();
    }

    class Boss : Monster
    {
        public override void Atk()
        {
            Console.WriteLine("Boss攻击");
        }
    }

    class Goblin : Monster
    {
        public override void Atk()
        {
            Console.WriteLine("Goblin攻击");
        }
    }


    internal class Program
    {
        static void Main(string[] args)
        {
            Boss boss1 = new Boss();
            Boss boss2 = new Boss();
            Goblin goblin1 = new Goblin();
            Goblin goblin2 = new Goblin();
            for (int i = 0; i < Monster.monsters.Count; i++)
            {
                Monster.monsters[i].Atk();
            }
        }
    }
}
