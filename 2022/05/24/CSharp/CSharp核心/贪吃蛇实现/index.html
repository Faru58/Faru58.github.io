贪吃蛇实现

试着在工程里实现多脚本，一个类一个脚本

UML类图：面向对象程序开发时理清对象关系，养成面向对象编程习惯
关联，直接关联，聚合，依赖关系，复合

七大原则：

总体要实现高内聚，低耦合
使程序模块的可重用性、移植性增强

高内聚，低耦合：

1.类的角度：减少类内部，对其他类的调用
2.功能块的角度：减少模块之间的交互复杂度

单一职责原则（SPR-Single Responsibility Principle）

类被修改的概率很大，因此应该专注于单一的功能。
如果把多个功能放在同一个类中，功能之间就形成了关联，改变其中一个功能，有可能中止另一个功能

开闭原则（OCP-Open Closed Principle）

对扩展开放，对修改关闭
拓展开放：模块的行为可以被拓展从而满足新的需求
修改关闭：不允许修改模块的源代码（或尽量使修改最小化）

里式替换原则（LSP-Liskov Substitution Principle）

任何父类出现的地方，子类都可以代替

依赖倒转原则（DIP-Dependence Inversion Principle）

要依赖于抽象（接口），不要依赖于具体的实现

迪米特原则（LoD-Law of Demeter）

又称最小知识原则
一个对象应该对其他对象尽可能少的了解
一个对象中的成员，要尽可能少的直接和其他类建立关系，目的是降低耦合性

接口分离原则（ISP-Interface Segregation Principle）

一个接口不需要提供太多的行为，一个接口应该尽量只提供一个对外的功能，让别人去选择需要实现什么样的行为，而不是把所有的行为封装到一个接口里

合成复用原则（CRP-Composite Reuse Principle）

尽量使用对象组合，而不是继承来达到复用的目的
继承是强耦合，组合关系是弱耦合
注意：不能盲目的使用合成复用原则，要在遵循迪米特原则的前提下

在开始做项目之前，整理UML类图时，
先按自己的想法把需要的类整理出来
再基于七大法则去优化整理自己的设计
最后，具体问题具体分析，面向对象的编程能力提升是需要经验积累的。

BeginOrEndBaseScene

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 贪吃蛇.Scene
{
    //抽象基类
    abstract class BeginOrEndBaseScene:ISceneUpdate
    {
        protected int nowSelIndex;
        protected string strTitle;
        protected string strOne;

        public abstract void EnterJDoSomething();

        public void Update()
        {
            Console.ForegroundColor = ConsoleColor.White;
            //显示标题
            Console.SetCursorPosition( Game.w / 2 - strTitle.Length, 5);
            Console.WriteLine(strTitle);


            //显示下方选项
            Console.SetCursorPosition(Game.w / 2 - strOne.Length, 8);
            Console.ForegroundColor = nowSelIndex == 0 ? ConsoleColor.Red : ConsoleColor.White;
            Console.WriteLine(strOne);
            Console.SetCursorPosition(Game.w / 2 - 4, 10);
            Console.ForegroundColor = nowSelIndex == 1 ? ConsoleColor.Red : ConsoleColor.White;
            Console.WriteLine("结束游戏");

            //检测输入
            switch(Console.ReadKey(true).Key)
            {
                case ConsoleKey.W:
                    --nowSelIndex;
                    if(nowSelIndex < 0)
                    {
                        nowSelIndex = 0;
                    }
                    break;
                case ConsoleKey.S:
                    ++nowSelIndex;
                    if (nowSelIndex > 1)
                    {
                        nowSelIndex = 1;
                    }
                    break;
                case ConsoleKey.J:
                    EnterJDoSomething();
                    break;
            }

        }
    }
}

GameScene

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using 贪吃蛇.Snake;

namespace 贪吃蛇.Scene
{
    class GameScene: ISceneUpdate
    {
        MapObject map;
        SnakeObject snake;
        Food food;
        //延迟计数
        int updateIndex = 0;

        public GameScene()
        {
            map = new MapObject();
            snake = new SnakeObject(40, 10);
            food = new Food(snake);
            map.Draw();
        }

        public void Update()
        {
            if (updateIndex % 5555 == 0)
            {

                food.Draw();

                snake.Move();
                snake.Draw();

                //检测是否撞墙
                if (snake.CheckEnd(map))
                {
                    //结束逻辑
                    Game.ChangeScene(E_SceneType.End);
                }

                snake.CheckEatFood(food);

                updateIndex = 0;
            }
            ++updateIndex;

            //在控制台中 检测玩家输入 让程序不被检测卡主
            //判断 有没有键盘输入 如果有 才为true
            if (Console.KeyAvailable)
            {
                //检测输入输出 不能再 间隔帧里面去处理 应该每次都检测 这样才准确
                switch (Console.ReadKey(true).Key)
                {
                    case ConsoleKey.W:
                        snake.ChangeDir(E_MoveDir.Up);
                        break;
                    case ConsoleKey.A:
                        snake.ChangeDir(E_MoveDir.Left);
                        break;
                    case ConsoleKey.S:
                        snake.ChangeDir(E_MoveDir.Down);
                        break;
                    case ConsoleKey.D:
                        snake.ChangeDir(E_MoveDir.Right);
                        break;
                }
            }
        }
    }
}

Game

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 贪吃蛇.Scene
{
    /// <summary>
    /// 场景类型枚举
    /// </summary>
    enum E_SceneType
    {
        /// <summary>
        /// 开始场景
        /// </summary>
        Begin,
        /// <summary>
        /// 游戏场景
        /// </summary>
        Game,
        /// <summary>
        /// 结束场景
        /// </summary>
        End,
    }

    class Game
    {
        //游戏窗口宽高
        public const int w = 80;
        public const int h = 20;
        //当前选中的场景
        public static ISceneUpdate nowScene;

        //初始化的方法
        public Game()
        {
            Console.CursorVisible = false;
            Console.SetWindowSize(w, h);
            Console.SetBufferSize(w, h);
            ChangeScene(E_SceneType.Begin);
        }

        //游戏开始的方法
        public void Start()
        {
            //游戏主循环 负责游戏场景的更新
            while(true)
            {
                //判断当前游戏场景不为空 就更新
                if(nowScene != null)
                {
                    nowScene.Update();
                }
            }
        }

        //切换场景的方法
        public static void ChangeScene(E_SceneType type)
        {
            //切场景之前，应该清空上一个场景的内容
            Console.Clear();

            switch (type)
            {
                case E_SceneType.Begin:
                    nowScene = new BeginScene();
                    break;
                case E_SceneType.Game:
                    nowScene = new GameScene();
                    break;
                case E_SceneType.End:
                    nowScene = new EndScene();
                    break;
            }
        }
    }
}

Position

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 贪吃蛇.Snake
{
    struct Position
    {
        public int x;
        public int y;

        public Position(int x, int y)
        {
            this.x = x;
            this.y = y;
        }

        //贪吃蛇中存在位置的比较(头和身体、事物、墙比较)
        public static bool operator ==(Position p1, Position p2)
        {
            if(p1.x == p2.x && p1.y == p2.y)
            {
                return true;
            }
            else
            {
                return false;
            }       
        }
        public static bool operator !=(Position p1, Position p2)
        {
            if (p1.x == p2.x && p1.y == p2.y)
            {
                return false;
            }
            else
            {
                return true;
            }
        }
    }
}

    /// 蛇身体类型
    /// </summary>
    enum E_SnakeBody_Type
    {
        /// <summary>
        /// 蛇头
        /// </summary>
        Head,
        /// <summary>
        /// 蛇躯
        /// </summary>
        Body,
    }

    class SnakeBody:GameObject
    {
        private E_SnakeBody_Type type;
        public SnakeBody(E_SnakeBody_Type type, int x, int y)
        {
            this.type = type;
            this.pos = new Position(x, y);
        }

        public override void Draw()
        {
            Console.SetCursorPosition(pos.x, pos.y);
            Console.ForegroundColor = type == E_SnakeBody_Type.Head ? ConsoleColor.Yellow : ConsoleColor.Green;
            Console.Write(type == E_SnakeBody_Type.Head ? "●" : "◎");
        }
    }
}

MapObject

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using 贪吃蛇.Scene;

namespace 贪吃蛇.Snake
{
    class MapObject : IDraw
    {
        public Wall[] walls;
        
        public MapObject()
        {
            walls = new Wall[ Game.w + (Game.h - 3) * 2 ];
            int index = 0;
            for (int i = 0; i < Game.w; i += 2)
            {
                //上面的墙
                walls[index++] = new Wall(i, 0);
            }
            for (int i = 0; i < Game.w; i += 2)
            {
                //下面的墙
                walls[index++] = new Wall(i, Game.h - 2);
            }
            for (int i = 1; i < Game.h - 2; i++)
            {
                //左边的墙
                walls[index++] = new Wall(0, i);
            }
            for (int i = 1; i < Game.h - 2; i++)
            {
                //右边的墙
                walls[index++] = new Wall(Game.w - 2, i);
            }
        }

        public void Draw()
        {
            for (int i = 0; i < walls.Length; i++)
            {
                walls[i].Draw();
            }
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using 贪吃蛇.Scene;

namespace 贪吃蛇.Snake
{
    class Food:GameObject
    {
        public Food(SnakeObject snake)
        {
            RandomPos(snake);
        }

        public override void Draw()
        {
            Console.SetCursorPosition(pos.x, pos.y);
            Console.ForegroundColor = ConsoleColor.Cyan;
            Console.Write("¤");
        }

        //随机位置的行为 行为 和蛇的位置 有关系 有了蛇再来考虑
        public void RandomPos(SnakeObject snake)
        {
            //随机位置
            Random r = new Random();
            int x = r.Next(2, Game.w / 2 - 1) * 2;
            int y = r.Next(1, Game.h - 4);
            pos = new Position(x, y);
            //得到蛇
            //如果重合 就会进if语句
            if (snake.CheckSamePos(pos))
            {
                RandomPos(snake);
            }
        }
    }
}

SnakeObject

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 贪吃蛇.Snake
{
    /// <summary>
    /// 蛇的移动方向
    /// </summary>
    enum E_MoveDir
    {
        /// <summary>
        /// 向上
        /// </summary>
        Up,
        /// <summary>
        /// 向下
        /// </summary>
        Down,
        /// <summary>
        /// 向左
        /// </summary>
        Left,
        /// <summary>
        /// 向右
        /// </summary>
        Right,
    }

    class SnakeObject : IDraw
    {
        SnakeBody[] bodys;
        //记录蛇的长度
        int nowNum;
        //定义初始移动方向
        E_MoveDir dir;

        //构造函数定义蛇
        public SnakeObject(int x, int y)
        {
            //粗暴定义，最多100
            bodys = new SnakeBody[200];
            //蛇头
            bodys[0] = new SnakeBody(E_SnakeBody_Type.Head, x, y);
            //之后身子从1开始
            nowNum = 1;
            dir = E_MoveDir.Right;
        }

        //蛇绘制
        public void Draw()
        {
            for (int i = 0; i < nowNum; i++)
            {
                bodys[i].Draw();
            }
        }

        //蛇移动
        public void Move()
        {
            //移动前,擦除最后一个位置
            SnakeBody lastBody = bodys[nowNum - 1];
            Console.SetCursorPosition(lastBody.pos.x, lastBody.pos.y);
            Console.Write("  ");

            //在蛇头移动之前 从蛇尾开始 不停的 让后一个的位置 等于前一个的位置
            for (int i = nowNum - 1; i > 0; i--)
            {
                bodys[i].pos = bodys[i - 1].pos;
            }

            //再动
            switch (dir)
            {
                case E_MoveDir.Up:
                    --bodys[0].pos.y;
                    break;
                case E_MoveDir.Down:
                    ++bodys[0].pos.y;
                    break;
                case E_MoveDir.Left:
                    bodys[0].pos.x -= 2;
                    break;
                case E_MoveDir.Right:
                    bodys[0].pos.x += 2;
                    break;
            }
        }

        //蛇改变移动方向
        public void ChangeDir(E_MoveDir dir)
        {
            //只有头时，可以随意移动；有身体时，不能向相反方向移动
            //判断何时不能转向
            if (dir == this.dir ||
                nowNum > 1 &&
                (this.dir == E_MoveDir.Up && dir == E_MoveDir.Down ||
                this.dir == E_MoveDir.Left && dir == E_MoveDir.Right ||
                this.dir == E_MoveDir.Down && dir == E_MoveDir.Up ||
                this.dir == E_MoveDir.Right && dir == E_MoveDir.Left)
            )
            {
                return;
            }
            this.dir = dir;
        }

        //蛇吃食物
        public void CheckEatFood(Food food)
        {
            if (bodys[0].pos == food.pos)
            {
                //吃到了 就应该让食物 位置再随机 增加蛇身体的长度
                food.RandomPos(this);
                //长身体
                AddBody();
            }
        }

        //蛇增加长度(与删除类似)
        private void AddBody()
        {
            SnakeBody frontBody = bodys[nowNum - 1];
            //先长 
            bodys[nowNum] = new SnakeBody(E_SnakeBody_Type.Body, frontBody.pos.x, frontBody.pos.y);
            //再加长度
            nowNum++;
        }

        //定义一个判断是否重合的方法用于食物
        public bool CheckSamePos(Position p)
        {
            for (int i = 0; i < nowNum; i++)
            {
                if (bodys[i].pos == p)
                {
                    return true;
                }
            }
            return false;
        }

        //蛇死亡（蛇头碰到墙壁或者身体）
        public bool CheckEnd(MapObject map)
        {
            //是否和墙体位置重合
            for (int i = 0; i < map.walls.Length; i++)
            {
                if (bodys[0].pos == map.walls[i].pos)
                {
                    return true;
                }
            }
            //是否碰到身体
            for (int i = 1; i < nowNum; i++)
            {
                if (bodys[0].pos == bodys[i].pos)
                {
                    return true;
                }
            }
            return false;
        }
    }
}
