运算符重载

运算符重载

基本概念

让自定义类和结构体，能够使用运算符
关键词 operator

特点：
1.一定是一个公共的静态方法
2.返回值写在operator前面
3.逻辑处理自定义

作用：
让自定义类和结构体对象可以进行运算

注意：
1.条件运算符需要成对实现
2.一个符号可以多个重载
3.不能使用ref和out

基本语法


class Point
{
    public int x;
    public int y;

    //重载+号
    public static Point operator +(Point p1, Point p2)
    //public static 返回类型 operator 运算符(参数列表)
    {
        Point p = new Point();
        p.x = p1.x + p2.x;
        p.y = p1.y + p2.y;
        return p;
    }

    //+号多个重载
    public static Point operator +(Point p1, int value)
    //public static 返回类型 operator 运算符(参数列表)
    {
        Point p = new Point();
        p.x = p1.x + value;
        p.y = p1.y + value;
        return p;
    }
}


internal class Program
{
    static void Main(string[] args)
    {
        Point p1 = new Point();
        p1.x = 1;
        p1.y = 1;
        Point p2 = new Point();
        p2.x = 2;
        p2.y = 2;

        Point p3 = p1 + p2;
        Console.WriteLine("({0},{1})", p3.x, p3.y);

        Point p4 = p3 + 2; //顺序不能错 一定是先Point类,再int类
        Console.WriteLine("({0},{1})", p4.x, p4.y);
    }
}

可重载运算符

算术运算符

+ - * / %  ++ --
注意: 符号需要两个参数还是一个参数

逻辑运算符

//!(逻辑非)

位运算符

| & ^ ~ << >>
//注意: 符号需要两个参数还是一个参数

条件运算符

//> < => <= == !=

//2.相关符号必须配对实现(实现了>，就要同时实现<)

不可重载运算符

//逻辑与  && 
//逻辑或  ||

练习题

题1

using System;

namespace Lesson10_运算符重载
{
    class Vector2
    {
        public int x;
        public int y;

        //重载==号
        public static bool operator ==(Vector2 v1, Vector2 v2)
        {
            return v1.x == v2.x && v1.y == v2.y;
        }
        public static bool operator !=(Vector2 v1, Vector2 v2)
        {
            return !(v1.x == v2.x && v1.y == v2.y);
        }
    }


    internal class Program
    {
        static void Main(string[] args)
        {
            Vector2 v1 = new Vector2();
            v1.x = 1;
            v1.y = 1;
            Vector2 v2 = new Vector2();
            v2.x = 1;
            v2.y = 1;

            Console.WriteLine(v1 == v2);     
        }
    }
}

题2

using System;

namespace Lesson10_运算符重载
{
    class Vector3
    {
        public int x;
        public int y;
        public int z;

        //重载==号
        public static Vector3 operator +(Vector3 v1, Vector3 v2)
        {
            Vector3 v = new Vector3();
            v.x = v1.x + v2.x;
            v.y = v1.y + v2.y;
            v.z = v1.z + v2.z;
            return v;
        }
        public static Vector3 operator -(Vector3 v1, Vector3 v2)
        {
            Vector3 v = new Vector3();
            v.x = v1.x - v2.x;
            v.y = v1.y - v2.y;
            v.z = v1.z - v2.z;
            return v;
        }
        public static Vector3 operator *(Vector3 v1, int num)
        {
            Vector3 v = new Vector3();
            v.x = v1.x * num;
            v.y = v1.y * num;
            v.z = v1.z * num;
            return v;
        }
    }


    internal class Program
    {
        static void Main(string[] args)
        {
            Vector3 v1 = new Vector3();
            v1.x = 1;
            v1.y = 1;
            v1.z = 1;

            Vector3 v2 = new Vector3();
            v2.x = 2;
            v2.y = 3;
            v2.z = 4;

            int num = 8;

            Vector3 v3 = v1 + v2;
            Vector3 v4 = v2 - v1;
            Vector3 v5 = v2 * num;

            Console.WriteLine("({0},{1},{2})", v3.x, v3.y, v3.z);
            Console.WriteLine("({0},{1},{2})", v4.x, v4.y, v4.z);
            Console.WriteLine("({0},{1},{2})", v5.x, v5.y, v5.z);
        }
    }
}
