2022.5.3

递归函数

重点知识点：
递归函数
概念： 让函数自己调用自己
注意：
一个正确的递归函数
1.必须有结束调用的条件
2.用于条件判断的 这个条件必须改变 能够打到停止的目的

练习题

using System;

namespace Lesson11_递归函数练习题
{
    internal class Program
    {
        #region 练习题1
        static void ZeroTen(int a)
        {
            if(a > 10)
            {
                return;
            }
            Console.WriteLine(a);
            ZeroTen(++a);
        }

        #endregion

        #region 练习题2
        static int Factorial(int a)
        {
            if(a == 1)
            {
                return 1;
            }
            return a * Factorial(--a);//--a应写在后面, 或用a - 1
            //若是 Factorial(--a) * a ,后面的a是已经减掉的a,得不到正确结果
        }

        #endregion

        #region 练习题3
        static long SumFactorial(int a)
        {
            if(a == 1)
            {
                return 1;
            }
            return Factorial(a) + SumFactorial(--a);
        }
        #endregion

        #region 练习题4
        //自己写的
        static double ZhuganLength1(int a)
        {
            if(a == 0)
            {
                return 100;
            }
            return 0.5 * ZhuganLength1(--a);
        }

        //答案
        //一是长度 二是天数
        static void ZhuganLength2(double length, int day = 0)
        {
            length /= 2;
            ++day;
            //第十天砍了后的长度
            if (day == 10)
            {
                Console.WriteLine("第十天砍后竹竿长{0}米", length);
                return;
            }            
            ZhuganLength2(length, day);
        }

        #endregion

        #region 练习题5
        //看答案的 想不出来短路怎么写
        static bool PrintNum(int a)
        {
            Console.WriteLine(a);           
            return a == 200 || PrintNum(a + 1); 
            //逻辑短路，若逻辑或前面的为真，不执行逻辑或后面的
        }
        
        #endregion


        static void Main(string[] args)
        {
            Console.WriteLine("递归函数练习题");

            //ZeroTen(0);

            Console.WriteLine(Factorial(5));

            Console.WriteLine(SumFactorial(10));

            Console.WriteLine(ZhuganLength1(10));
            ZhuganLength2(100, 0);

            PrintNum(1);
        }
    }
}
