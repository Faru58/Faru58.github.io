结构体

2022.5.3

#结构体

重点知识点：
结构体
1.概念： 结构体 struct 是变量和函数的集合 用来表示特定的数据集合
2.访问修饰符：用来修饰变量和方法 public->外部可以调用 private->只能内部调用 不写默认private
3.构造函数：没有返回值 函数名和结构体名相同 可以重载 主要是帮助我们快速初始化结构体对象

注意：
1.在结构体中声明的变量不能初始化 只能在外部或者函数中赋值（初始化）
2.在结构体中声明的函数 不用加static

#练习题

以下是答案：

using System;

namespace Lesson12_结构体练习题
{
    #region 练习题1
    struct Student
    {       
        public string name;
        public bool sex;
        public int age;
        public int lesson;
        public string major;
        public Student(string name, bool sex, int age, int lesson, string major)
        {
            this.name = name;           
            this.sex = sex;
            this.age = age;
            this.lesson = lesson;
            this.major = major;
        }
        public void Information()
        {
            Console.WriteLine("学员姓名{0},性别{1},年龄{2},班级{3},专业{4}", name, age, age, lesson, major);
        }
    }
    #endregion

    #region 练习题2
    //他们都是访问修饰符 用来修饰变量和方法
    //private 只能内部调用
    //public  外部可以调用
    #endregion

    #region 练习题3
    struct Rectangular
    {
        public int x;
        public int y;
        public Rectangular(int x, int y)
        {
            this.x = x;
            this.y = y;
        }
        public void Information()
        {
            Console.WriteLine("矩形的长为{0},宽为{1},面积为{2},周长为{3}", x, y, x * y, 2 *(x + y));
        }
    }
    #endregion

    #region 练习题4
    struct PlayerInfo
    {
        public string name;
        public E_Occupation occupation;        
        public PlayerInfo(string name, E_Occupation occupation)
        {
            this.name = name;
            this.occupation = occupation;        
        }
        public void AttackInformation()
        {
            string o = "";
            string s = "";
            switch (occupation)
            {
                case E_Occupation.Warrior:
                    o = "战士";
                    s = "冲锋";
                    break;
                case E_Occupation.Hunter:
                    o = "猎人";
                    s = "假死";
                    break;
                case E_Occupation.Witch:
                    o = "法师";
                    s = "奥术冲击";
                    break;
            }
            Console.WriteLine("{0}{1}施放了{2}", o, name, s);
        }
    }
    enum E_Occupation
    {
        Warrior,
        Hunter,
        Witch,
    }
    #endregion

    #region 练习题5
    struct Monster
    {
        public string name;
        public int atk;
        public Monster(string name)
        {
            this.name = name;
            Random r = new Random();
            atk = r.Next( 10, 30 );
        }
        public void Atk()
        {
            Console.WriteLine("{0}的攻击力是{1}", name, atk);
        }
    }
    #endregion

    #region 练习题7
    struct UltraMan
    {
        public string name;
        public int atk;
        public int def;
        public int hp;
        public UltraMan(string name, int atk, int def, int hp)
        {
            this.name = name;
            this.atk = atk;
            this.def = def;
            this.hp = hp;           
        }

        //结构体是值类型 想要在函数内部改变值类型信息 外部受影响 一定要用ref或out
        public void Attack(ref Boss monster)
        {
            //奥特曼打怪兽的逻辑
            monster.hp -= atk - monster.def;
            Console.WriteLine("{0}攻击了{1}，造成了{2}点伤害，{1}剩余血量{3}", name, monster.name, atk - monster.def, monster.hp);
        }
    }
    struct Boss
    {
        public string name;
        public int atk;
        public int def;
        public int hp;
        public Boss(string name, int atk, int def, int hp)
        {
            this.name = name;
            this.atk = atk;
            this.def = def;
            this.hp = hp;
        }
        public void Attack(ref UltraMan Tiga)
        {
            //怪兽打奥特曼的逻辑
            Tiga.hp -= atk - Tiga.def;
            Console.WriteLine("{0}攻击了{1}，造成了{2}点伤害，{1}剩余血量{3}", name, Tiga.name, atk - Tiga.def, Tiga.hp);
        }
    }
    #endregion


    internal class Program
    {
        
        
        static void Main(string[] args)
        {
            Console.WriteLine("结构体练习题");

            #region 练习题1
            Student s1 = new Student("季宝", true, 23, 2, "微电子");
            Student s2 = new Student("大逼", true, 23, 2, "网安");
            s1.Information();
            s2.Information();
            #endregion

            #region 练习题3
            Rectangular r1 = new Rectangular (5, 4);
            r1.Information();
            #endregion

            #region 练习题4
            Console.Write("请输入玩家名字：");
            string name = Console.ReadLine();
            Console.Write("请选择你的职业：（0战士，1猎人，2法师）");
            try 
            {
                E_Occupation o = (E_Occupation)int.Parse(Console.ReadLine());
                PlayerInfo p1 = new PlayerInfo(name ,o);
                p1.AttackInformation();            }
            catch
            {
                Console.WriteLine("请输入数字");
            }
            #endregion

            #region 练习题6
            //声明Monster数组
            Monster[] monsters = new Monster[10];
            for (int i = 0; i < monsters.Length; i++)
            {
                monsters[i] = new Monster("怪兽" + i);
                monsters[i].Atk();
            }

            #endregion

            #region 练习题7
            UltraMan Tiga = new UltraMan("迪迦", 10, 5, 100);
            Boss boss = new Boss("哥斯拉", 8, 4, 100);
            while(true)
            {
                Tiga.Attack(ref boss);
                if( boss.hp <= 0)
                {
                    Console.WriteLine("迪迦胜利");
                    break;
                }
                boss.Attack(ref Tiga);
                if (Tiga.hp <= 0)
                {
                    Console.WriteLine("哥斯拉胜利");
                    break;
                }
                Console.WriteLine("按任意键继续");
                Console.ReadKey(true);
            }
            #endregion
        }
    }
}
