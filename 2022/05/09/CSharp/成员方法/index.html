成员方法

成员方法

成员方法的声明

基本概念：
成员方法（函数） 用来表现对象行为
1.声明在语句块中
2.是用来描述对象的行为的
3.规则和函数声明规则相同
4.受到访问修饰符规则影响
5.返回值参数不做限制
6.方法数量不做限制

注意：
1.成员方法不要加static关键字
2.成员方法必须实例化出对象，再通过对象来使用，相当于该对象执行了某个行为
3.成员方法 受到访问修饰符影响

成员方法的使用

必须实例化出对象，再通过对象来使用，相当于该对象执行了某个行为

练习题

using System;

namespace Lesson3_成员方法
{
    #region 练习题1
    class Person
    {
        public string name;

        /// <summary>
        /// 说话
        /// </summary>
        /// <param name="str"></param>
        public void Speak(string str)
        {
            Console.WriteLine("{0}说{1}", name, str);
        }

        /// <summary>
        /// 走路
        /// </summary>
        /// <param name="s"></param>
        public void Walk(int s)
        {
            Console.WriteLine("{0}走了{1}步", name, s);
        }

        /// <summary>
        /// 吃饭
        /// </summary>
        /// <param name="food"></param>
        public void Eat(Food food)
        {
            Console.WriteLine("{0}吃了{1},吸收了{2}热量", name, food.name, food.heat);
        }
    }
    #endregion

    #region 练习题2
    class Student
    {
        public string name;

        /// <summary>
        /// 学习
        /// </summary>
        /// <param name="subject"></param>
        public void Study(string subject)
        {
            Console.WriteLine("{0}在学习{1}", name, subject);
        }

        /// <summary>
        /// 吃饭
        /// </summary>
        /// <param name="food"></param>
        public void Eat(Food food)
        {
            Console.WriteLine("{0}吃了{1},吸收了{2}热量", name, food.name, food.heat);
        }
    }
    #endregion

    #region 练习题3
    class Food
    {
        public E_FoodName name;
        public int heat;
    }
    /// <summary>
    /// 食物种类枚举
    /// </summary>
    enum E_FoodName
    {
        Apple = 10,
        Banana = 20,
        Mango = 30,
    }
    #endregion
    
    internal class Program
    {


        static void Main(string[] args)
        {
            #region 练习题123
            Person p = new Person();
            p.name = "大逼";

            Student s = new Student();
            s.name = "季宝";

            Food f = new Food();
            f.name = E_FoodName.Apple;
            f.heat = (int)f.name;

            p.Eat(f);
            s.Eat(f);
            #endregion
        }
    }
}
