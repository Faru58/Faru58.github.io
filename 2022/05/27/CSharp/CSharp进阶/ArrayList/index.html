ArrayList

ArrayList是C#已经封装好的一个类
它的本质是一个object类型的数组

声明

using System.Collections;
    ArrayList array = new ArrayList();

增

//增加一个元素
array.Add(object);
//批量增加，把另一个list容器中的元素加到后面
array.AddRange(array2);
//插入元素
array.Insert(索引值, object); //在该位置插入想要插入的元素

删

//移出指定元素，从头找，找到删
array.Remove(object);
//移除指定位置元素
array.RemoveAt(索引值);
//清空数组
array.Clear();

查

//得到指定位置的元素
Console.WriteLine(array[0]);
//查询元素是否存在
array.Contains(object); //存在为真,不存在为假
//正向查找元素位置,正向第一个找到的位置
array.IndexOf(object); //返回索引值,没找到返回-1 
//正向查找元素位置,反向第一个找到的位置
array.LastIndexOf(object); //返回索引值,没找到返回-1

改

//直接改指定位置的元素
array[索引值] = object;

遍历

//数组长度
array.Count
//数组容量
array.Capacity
//遍历
for( i = 0; i < array.Count; i++ )
{
    //需要的逻辑
}
//迭代器遍历 把array里的元素放到item里
foreach (object item in array)
{
    //需要的逻辑 
}

装箱拆箱

ArrayList本质上是一个可以自动扩容的object数组
由于用万物之父object来存储数据，自然存在装箱拆箱。
当往其中进行值类型存储时即为装箱，当将值类型对象取出来时即为拆箱
ArrayList尽量少用

int i = 1;
array[0] = i; //装箱
i = (int)array[0]; //拆箱

题2

using System;
using System.Collections;

namespace Lesson1_ArrayList
{
    /// <summary>
    /// 道具种类
    /// </summary>
    enum E_ItemType
    {
        /// <summary>
        /// 回血药
        /// </summary>
        HpRecovery,
        /// <summary>
        /// 回蓝药
        /// </summary>
        MpRecovery,
        /// <summary>
        /// 复活药
        /// </summary>
        Revival,
    }

    class BagManagement
    {
        //初始金币100
        private int money = 100;
        //道具
        private ArrayList items = new ArrayList();

        public BagManagement()
        {
            items.Clear();
        }
        //购买道具方法
        public void BuyItem(Item i)
        {
            if((money - i.price) > 0 )
            {
                items.Add(i);
                money -= i.price;
                Console.WriteLine("购买成功\n");
            }
            else
            {
                Console.WriteLine("金币不够，购买失败\n");
            }              
        }

        //卖出道具方法
        public void SellItem(Item i)
        {
            if(items.Contains(i))
            {
                items.Remove(i);
                money += i.price;
                Console.WriteLine("出售成功\n");
            }
            else
            {
                Console.WriteLine("出售失败\n");
            }
            
        }

        //显示道具方法
        public void MyBag()
        {
            if(items.Count > 0)
            {
                for (int i = 0; i < items.Count; i++)
                {
                    (items[i] as Item).Show(); 
                }
                Console.WriteLine("现在还有{0}金币\n", money);
            }
            else
            {
                Console.WriteLine("背包为空");
            }
        }

    }

    abstract class Item
    {
        abstract public void Show();
        protected E_ItemType type;
        public int price;
    }

    class HpRecovery : Item
    {
        public HpRecovery()
        {
            type = E_ItemType.HpRecovery;
            price = 20;
        }

        public override void Show()
        {
            Console.Write("回血药*1,");
        }
    }

    class MpRecovery : Item
    {
        public MpRecovery()
        {
            type = E_ItemType.MpRecovery;
            price = 10;
        }

        public override void Show()
        {
            Console.Write("回蓝药*1,");
        }
    }

    class Revival : Item
    {
        public Revival()
        {
            type = E_ItemType.Revival;
            price = 50;
        }

        public override void Show()
        {
            Console.Write("复活药*1,");
        }
    }



    internal class Program
    {
        static void Main(string[] args)
        {
            Item hpRecovery = new HpRecovery();
            Item mpRecovery = new MpRecovery();
            Item Revival = new Revival();
            BagManagement bagManagement = new BagManagement();
            Console.WriteLine("进入商店，店里有回血药（20金币）、回蓝药（10金币）、复活药（50）金币");
            
            Console.WriteLine("");

            while (true)
            {
                
                try
                {
                    Console.WriteLine("按q进入购买模式，按w进入出售模式，按e显示背包，按r退出商店");
                    switch (Console.ReadKey(true).Key)
                    {
                        case ConsoleKey.Q:
                            Console.WriteLine("按a选择回血药，按s选择回蓝药，按d选择复活药");
                            try
                            {
                                switch (Console.ReadKey(true).Key)
                                {
                                    case ConsoleKey.A:
                                        bagManagement.BuyItem(hpRecovery);
                                        break;
                                    case ConsoleKey.S:
                                        bagManagement.BuyItem(mpRecovery);
                                        break;
                                    case ConsoleKey.D:
                                        bagManagement.BuyItem(Revival);
                                        break;
                                }
                            }
                            catch { Console.WriteLine("请输入正确指令\n"); }
                            break;                        
                        case ConsoleKey.W:
                            Console.WriteLine("按a选择回血药，按s选择回蓝药，按d选择复活药");
                            try
                            {
                                switch (Console.ReadKey(true).Key)
                                {
                                    case ConsoleKey.A:
                                        bagManagement.SellItem(hpRecovery);
                                        break;
                                    case ConsoleKey.S:
                                        bagManagement.SellItem(mpRecovery);
                                        break;
                                    case ConsoleKey.D:
                                        bagManagement.SellItem(Revival);
                                        break;
                                }
                            }
                            catch { Console.WriteLine("请输入正确指令\n"); }
                            break;
                        case ConsoleKey.E:
                            bagManagement.MyBag();
                            break;
                        case ConsoleKey.R:
                            Environment.Exit(0);
                            break;
                    }
                }
                catch
                {
                    Console.WriteLine("请输入正确指令\n");
                }
            }
        }
    }
}
