接口

概念

接口是行为的抽象规范，也是一种自定义类型
用于存储不同类的对象之间相同的方法
关键字：interface

声明规范：
1.不包含成员变量
2.只包含方法、属性、索引器、事件
3.成员不能被实现
4.成员可以不用写访问修饰符（默认public），不能是private
5.接口不能继承类，但是可以继承另一个接口

使用规范：
1.类可以继承多个接口
2.类继承接口后，必须实现接口中所有成员

接口的声明及使用

一句话记忆：接口是抽象行为的“基类”
接口命名规范：帕斯卡命名法前加I

接口用来继承：
1.类可以继承一个类,n个接口
2.继承了接口后,必须实现其中的内容,并且必须是public的
3.实现的接口函数,可以加virtual再写在子类里重写
4.接口也遵循里式替换原则

//interface 接口名
interface IFly
{
    //方法
    void Fly();

    //属性
    string Name
    {
        get; //不能有语句块,即默认自动属性
        set;
    }

    //索引器
    int this[int index]
    {
        get;
        set;
    }

    //事件
    event Action doSomething;
} 

class Animal
{

}
//1.类可以继承一个类,n个接口
class Person : Animal, Ifly
{
    //2.继承了接口后,必须实现其中的内容,并且必须是public的
    //3.实现的接口函数,可以加virtual再写在子类里重写
    public virtual void Fly()
    {

    }

    public string Name
    {
        get;
        set;
    }

    public int this[int index]
    {
        get;
        set;
    }

    public event Action doSomething;
}

internal class Program
{
    static void Main(string[] args)
    {
        //4.接口也遵循里式替换原则
        IFly f = new Person();
        //注意 IFly f = new IFly();是错误的,接口是不能实例化的
    }
}

接口可以继承接口

接口继承接口时,不需要实现
带类继承接口后,类自己要去实现所有内容

interface IWalk
{
    void Walk();
}

interface IMove : IWalk
{

}

class Test: IMove
{
    public void Walk();
}

显示实现接口

当一个类继承两个接口,但是接口中存在着同名方法时
注意：显示实现接口时,不能写访问修饰符

interface IAtk
{
    void Atk();
}
interface ISuperAtk
{
    void Atk();
}
class Player : IAtk, ISuperAtk
{
    //显示实现接口,就是用接口名.行为名来实现
    void IAtk.Atk()
    {
        
    }
    void ISuperAtk.Atk()
    {

    }
}

练习题

题1

using System;

namespace Lesson19_接口
{
    interface IRegister
    {
        void Register();
    }

    class Person : IRegister
    {
        public void Register()
        {
            Console.WriteLine("个人请到派出所登记");
        }
    }

    class Car : IRegister
    {
        public void Register()
        {
            Console.WriteLine("车辆请到车管所登记");
        }
    }

    class House : IRegister
    {
        public void Register()
        {
            Console.WriteLine("房子请到房管局登记");
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            //可用数组实例化
            IRegister p = new Person();
            p.Register();
            IRegister c = new Car();
            c.Register();
            IRegister h = new House();
            h.Register();
        }
    }
}

题2

using System;

namespace Lesson19_接口
{
    //定义三个接口 1.飞 2.游泳
    //定义一个鸟类(因为鸟类都能走，走就不用接口)

    //飞
    interface IFly
    {
        void Fly();
    }

    //游泳
    interface ISwim
    {
        void Swim();
    }


    //鸟类，鸟类都能走，用抽象类及抽象方法
    abstract class Bird
    {
        public abstract void Walk();
    }

    //麻雀，能飞，不能游泳
    class Sparrow : Bird, IFly
    {
        public void Fly()
        {
            Console.WriteLine("麻雀能飞");
        }

        public override void Walk()
        {
            Console.WriteLine("麻雀能走");
        }
    }

    //鸵鸟，不能飞，不能游泳
    class Ostrich : Bird
    {
        public override void Walk()
        {
            Console.WriteLine("鸵鸟能走");
        }
    }

    //企鹅，不能飞，能游泳
    class Penguin : Bird, ISwim
    {
        public void Swim()
        {
            Console.WriteLine("企鹅能游泳");
        }

        public override void Walk()
        {
            Console.WriteLine("企鹅能走");
        }
    }

    //鹦鹉，能飞，不能游泳
    class Parrot : Bird, IFly
    {
        public void Fly()
        {
            Console.WriteLine("鹦鹉能飞");
        }

        public override void Walk()
        {
            Console.WriteLine("鹦鹉能走");
        }
    }

    //天鹅，能飞，能游泳
    class Swan : Bird, IFly, ISwim
    {
        public void Fly()
        {
            Console.WriteLine("天鹅能飞");
        }

        public void Swim()
        {
            Console.WriteLine("天鹅能游泳");
        }

        public override void Walk()
        {
            Console.WriteLine("天鹅能走");
        }
    }

    //直升机
    class Helicopter : IFly
    {
        public void Fly()
        {
            Console.WriteLine("直升机能飞");
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            //请自行实例化
            Swan swan = new Swan();
            swan.Swim();
            swan.Walk();
            swan.Fly();
        }
    }
}

题3

本题关键：
public IUSB usb; 

using System;

namespace Lesson19_接口
{
    //定义接口 USB接口
    interface IUSB
    {
        void ReadData();
    }

    //存储设备类,插在USB接口上可以读取设备名
    class StorageDevice : IUSB
    {
        public string name;
        public StorageDevice(string name)
        {
            this.name = name;
        }

        public void ReadData()
        {
            Console.WriteLine("{0}正在传输数据", name);
        }
    }

    //MP3属于播放设备
    class Mp3 : IUSB
    {
        public void ReadData()
        {
            Console.WriteLine("MP3正在传输数据");
        }
    }

    //用于实例化的电脑类,它有一个USB接口
    class Computer
    {
        //声明一个接口成员，用于接口对接
        public IUSB usb;
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            Computer computer = new Computer();
            StorageDevice mhd = new StorageDevice("移动硬盘");
            StorageDevice usbd = new StorageDevice("U盘");
            Mp3 mp3 = new Mp3();

            //USB接移动硬盘
            computer.usb = mhd;
            computer.usb.ReadData();

            //USB接U盘
            computer.usb = usbd;
            computer.usb.ReadData();

            //USB接MP3
            computer.usb = mp3;
            computer.usb.ReadData();
        }
    }
}
