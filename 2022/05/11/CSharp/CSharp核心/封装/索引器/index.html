索引器

索引器

索引器

基本概念

让对象可以像数组一样通过索引访问其中元素
使程序看起来更直观，更容易编写

基本语法

class Person
{
    private string name;
    private int age;
    private Person[] friends;

    //访问修饰符 返回值 this[参数类型 参数名, 参数类型 参数名, ......]
    public Person this[int index]
    {
        get
        {
            return friends[index];
        }
        set
        {
            friends[index] = value;
        }
    }
}

索引器的使用

Person p = new Person();
p[0] = new Person();

索引器中可以写逻辑

class Person
{
    private string name;
    private int age;
    private Person[] friends;

    public Person this[int index]
    {
        get
        {
            if(friends == null || 
               friends.Length - 1 < index)
            {
                return null;
            }
            return friends[index];
        }
        set
        {
            if(friends == null)
            {
                friends = new Person[] { value };
            }
            else if(index > friends.Length - 1)
            {
                //自己定了一个规则 如果索引越界 就默认把最后一个朋友顶掉
                friends[friends.Length - 1] = value;
            }
            friends[index] = value;
        }
    }
}

索引器可以重载

重载的概念是：函数名相同，参数类型、数量、顺序不同

class Person
{
    private string name;
    private int age;
    private Person[] friends;

    private int[,] array;
    //重载this
    public int this[int i, int j]
    {
        get
        {
            return array[i, j];
        }
        set
        {
            array[i, j] = value;
        }
    }
    public int this[string str]
    {
        get
        {
            switch(str)
            {
                case "name":
                    return this.name;
                case "age":
                    return age.ToString();
            }
            return "";
        }
    }

    public Person this[int index]
    {
        get
        {
            return friends[index];
        }
        set
        {
            friends[index] = value;
        }
    }
}

总结

索引器的作用：
1.以中括号的形式访问自定义类中的元素
2.规则自己定，访问时与数组相同
3.比较适用于，在类中有数组变量时使用，可以方便地访问和进行逻辑处理

练习题

using System;

namespace Lesson6_索引器
{
    class Num
    {
        private int[] number = new int[5];

        //用索引器由外部定义数组
        public int this[int index]
        {
            get
            {
                return number[index];
            }
            set
            {
                number[index] = value;
            }
        }

        //增
        public void Add(Num arr, int num)
        {
            int[] number1 = new int[arr.number.Length + 1];
            for (int i = 0; i < arr.number.Length; i++)
            {
                number1[i] = arr.number[i];
            }
            number1[number.Length] = num;
            arr.number = number1;
            //因为number数组是private类型 只能在类内直接打印
            for (int i = 0; i < arr.number.Length; i++)
            {
                Console.WriteLine(arr.number[i]);
            }
        }

        //查
        public int Search(Num arr, int num)
        {
            //暴力遍历寻找
            for (int i = 0; i < arr.number.Length; i++)
            {
                if(arr.number[i] == num)
                {
                    //找到输出位置
                    return i;
                }
            }
            //找不到输出-1
            return -1;
        }

        //删
        public void Del(Num arr, int num)
        {     
            if(arr.Search(arr, num) != -1)
            {
                for (int i = arr.Search(arr, num); i < arr.number.Length - 1; i++)
                {
                    arr.number[i] = arr.number[i + 1];
                }
                arr.number[arr.number.Length - 1] = 0;
                Console.WriteLine("删除成功"); 
            }
            else
            {
                Console.WriteLine("没有您想要删的数，删除失败");
            }
        }

        //改  num1改成num2
        public void Change(Num arr, int num1, int num2)
        {
            if (arr.Search(arr, num1) != -1)
            {
                arr.number[arr.Search(arr, num1)] = num2;
                Console.WriteLine("修改成功");
            }
            else
            {
                Console.WriteLine("没有您想要改的数，修改失败");
            }
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            //初始化数组
            Num arr = new Num();
            arr[0] = 1;
            arr[1] = 15;
            arr[2] = 8;
            arr[3] = 11;
            arr[4] = 7;

            //增
            arr.Add(arr, 9);

            //删
            arr.Del(arr, 1);
            arr.Del(arr, 10);

            //查
            int n = int.Parse(Console.ReadLine());
            Console.WriteLine("您想找的数{0}在数组的第{1}位", n, arr.Search(arr, n) + 1);

            //改
            arr.Change(arr, 9, 66);
        }
    }
}
