角度和旋转知识点

角度和旋转知识点

角度和旋转

角度

相对世界坐标系角度

this.transform.eulerAngles

相对父坐标系角度

this.transform.localEulerAngles

注意：设置角度和设置位置一样，不能单独设置xyz，要一起设置

旋转

写在Update里

  1. 自转
    每个轴具体转多少度
    第一个参数：每一帧旋转的角度
    第二个参数：默认不填是相对于自己坐标系的旋转
//相对自己坐标轴
transform.Rotate(new Vector3(0, 10, 0) * Time.deltaTime);
//相对世界坐标轴
transform.Rotate(new Vector3(0, 10, 0) * Time.deltaTime, Space.World);
  1. 相对于某个轴转多少度
    第一个参数：相对哪个轴进行转动
    第二个参数：转动的角度是多少
    第三个参数：默认不填是相对于自己坐标系的旋转
//相对自己的坐标轴
transform.Rotate(Vector3.Right, 10 * Time.deltaTime);
//相对世界的坐标轴
transform.Rotate(Vector3.Right, 10 * Time.deltaTime, Space.World);
  1. 相对于某一个点转
    第一个参数：一个点
    第二个参数：想要绕的轴
    第三个参数：转动的角度
transform.RotateAround(Vector3.Zero, Vector3.Right, 10 * Time.deltaTime);

练习题

题1、2

创建一个空物体在炮管和炮台的连接处，让炮管作为空物体的子物体，则炮管会按着空物体的中心旋转

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Lesson7 : MonoBehaviour
{
    public float rotateSpeed = 10;
    public float headRotateSpeed = 10;
    public float connectRotateSpeed = 10;

    public Transform headTransform;
    public Transform connectTransform;
    // Start is called before the first frame update
    void Start()
    {
        //创建底座
        for (int i = 0; i < 2; i++)
        {
            Vector3 pos = new Vector3(-0.5f + i, -1.395f, 0.5f);
            for (int j = 0; j < 4; j++)
            {
                GameObject objCube = GameObject.CreatePrimitive(PrimitiveType.Cube);
                objCube.transform.position = pos + new Vector3(0, 0, -j);
                //让创建的底座成为Tank的子对象，使得可以一起旋转
                objCube.transform.parent = this.transform;
            }
        }
    }

    // Update is called once per frame
    void Update()
    {
        transform.Rotate(Vector3.up, rotateSpeed * Time.deltaTime);

        //炮台旋转
        headTransform.Rotate(Vector3.up, headRotateSpeed * Time.deltaTime);
        //通过localEulerAngles得到的角度只有0~360的正数
        if (!(headTransform.localEulerAngles.y >= 315 && headTransform.localEulerAngles.y <= 360)
            && (headTransform.localEulerAngles.y >= 45 && headRotateSpeed > 0))
        {
            headRotateSpeed = -headRotateSpeed;
        }
        else if(!(headTransform.localEulerAngles.y <= 45 && headTransform.localEulerAngles.y >= 0)
                && (headTransform.localEulerAngles.y <= 315 && headRotateSpeed < 0))
        {
            headRotateSpeed = -headRotateSpeed;
        }

        //炮管旋转
        connectTransform.Rotate(Vector3.right, connectRotateSpeed * Time.deltaTime);
        //通过localEulerAngles得到的角度只有0~360的正数
        if (!(connectTransform.localEulerAngles.x >= 345 && connectTransform.localEulerAngles.x <= 360)
            && (connectTransform.localEulerAngles.x >= 15 && connectRotateSpeed > 0))
        {
            connectRotateSpeed = -connectRotateSpeed;
        }
        else if (!(connectTransform.localEulerAngles.x <= 15 && connectTransform.localEulerAngles.x >= 0)
                && (connectTransform.localEulerAngles.x <= 345 && connectRotateSpeed < 0))
        {
            connectRotateSpeed = -connectRotateSpeed;
        }
    }
}

题3

月亮是地球的子对象，地球是太阳的子对象

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Lesson7_Planet : MonoBehaviour
{
    public Transform sun;
    public Transform earth;
    public Transform moon;
    // Start is called before the first frame update
    void Start()
    {
        
    }

    // Update is called once per frame
    void Update()
    {
        //自转
        sun.transform.Rotate(Vector3.up, 10 * Time.deltaTime);
        earth.transform.Rotate(Vector3.up, 10 * Time.deltaTime);
        moon.transform.Rotate(Vector3.up, 10 * Time.deltaTime);

        //公转
        earth.transform.RotateAround(sun.position, Vector3.up, 10 * Time.deltaTime);
        moon.transform.RotateAround(earth.position, Vector3.up, 10 * Time.deltaTime);
    }
}
