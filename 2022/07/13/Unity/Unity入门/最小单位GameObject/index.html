GameObject中的静态方法

unity中的Object不是C#里的万物之父，是unity自己的类

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class LessonGameObject : MonoBehaviour
{
    //准备用来克隆的对象
    //1. 直接是场景上的对象
    //2. 可以是一个预设体对象
    public GameObject myObj;

    // Start is called before the first frame update
    void Start()
    {
        //创建自带几何体
        //只要得到了一个GameObject对象，就可以通过GetComponent来得到它身上挂的任何脚本信息
        GameObject obj = GameObject.CreatePrimitive(PrimitiveType.Cube);
        obj.name = "xCube";

        //查找对象(无法找到失活对象)(存在多个同名或同tag对象，无法准确找到是哪个)
        //查找单个对象
        //1.通过对象名查找：效率较低，会在场景中所有对象中去查找，没有找到就会返回null
        GameObject obj2 = GameObject.Find("xCube");
        if(obj2 != null)
        {
            print(obj2.name);
        }
        //2.通过tag查找：两个方法一样
        //GameObject obj3 = GameObject.FindWithTag("Player");
        GameObject obj3 = GameObject.FindGameObjectWithTag("Player");
        if (obj3 != null)
        {
            print(obj3.name);
        }

        //查找多个对象(只能通过tag找，名字不行)
        GameObject[] objs = GameObject.FindGameObjectsWithTag("Player");
        print(objs.Length);

        //实例化(克隆对象)对象：根据一个GameObject对象，创建出一个和它一模一样的对象
        GameObject.Instantiate(myObj);

        //删除对象(第二个参数代表延迟几秒删除)
        GameObject.Destroy(obj, 5);
        //也可以删除脚本
        GameObject.Destroy(this);
        //注意：Destroy不会马上移除对象，是先给这个对象加一个移除标识，一般情况下会在下一帧移除对象并从内存中删除(异步)
        GameObject.DestroyImmediate(myObj); //立刻移除，但可能卡顿

        //过场景不移除，不想谁被移除就传谁
        //默认情况下，在切换场景时，场景中对象都会被自动删除掉
        GameObject.DontDestroyOnLoad(this.gameObject);
    }

    // Update is called once per frame
    void Update()
    {
        
    }
}

GameObject中的成员方法

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class LessonGameObject : MonoBehaviour
{
    // Start is called before the first frame update
    void Start()
    {
        //创建空物体
        GameObject obj6 = new GameObject();
        GameObject obj7 = new GameObject("空物体");
        GameObject obj8 = new GameObject("挂载脚本的空物体", typeof(LessonMonoBehaviour));

        //为对象添加脚本
        //继承MonoBehavior的脚本是不能new出来的
        //动态地添加继承MonoBehavior的脚本
        obj6.AddComponent(typeof(LessonMonoBehaviour));
        obj7.AddComponent<LessonMonoBehaviour>();

        //得到脚本 GetComponent

        //标签比较
        if (this.gameObject.CompareTag("Player"))
        {
            print("对象的标签 是 Player");
        }
        if (this.gameObject.tag == "Player")
        {
            print("对象的标签 是 Player");
        }

        //设置激活失活
        obj6.SetActive(false); //true激活 false失活


    }

    // Update is called once per frame
    void Update()
    {
        
    }
}

    实例化对象
    删除对象
    过场景不移除

  3. 成员方法
    为对象动态添加指定脚本
    获取对象上的脚本
    设置失活激活的状态

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

[SerializeField]
public enum E_Action
{
    Rename,
    DeActive,
    DelayDelete,
    ImmediatelyDelete,
}

public class Exam02A : MonoBehaviour
{
    public E_Action actiontype;
    public GameObject B;
    // Start is called before the first frame update
    void Start()
    {
        switch (actiontype)
        {
            case E_Action.Rename:
                B.name = "B改名";
                break;
            case E_Action.DeActive:
                B.SetActive(false);
                break;
            case E_Action.DelayDelete:
                Destroy(B, 1);
                break;
            case E_Action.ImmediatelyDelete:
                DestroyImmediate(B);
                break;
        }
    }

    // Update is called once per frame
    void Update()
    {
        
    }
}
