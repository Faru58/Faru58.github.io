List自带排序方法

系统自带的变量(int, double ,float, …)可以直接Sort

List<int> list = new List<int>();
//升序排序
list.Sort();
//ArrayList中也有Sort方法

自定义类的排序

class Item : IComparable<Item>
{
    public int money;

    public Item(int money)
    {
        this.money = money;
    }

    public int CompareTo(Item other)
    {
        //返回值的含义
        //大于0：放在传入对象的后面
        //等于0：保持当前位置不变
        //小于0：放在传入对象的前面
        if(this.money > other.money)    
        {
            return 1;
        }
        else 
        {
            return -1;
        }
    }
}

List<Item> listItem = new List<Item>();  
listItem.Sort();

通过委托函数进行排序

class shopItem 
{
    public int id;

    public shopItem(int id)
    {
        this.id = id;
    }
}

List<shopItem> listShopItem = new List<shopItem>();  
listShopItem.Sort(SortShopItem);

static int SortShopItem( shopItem a, shopItem b ) 
{
    //传入的两个对象为列表中的两个对象
    //左边的a和右边的b比较 a比b大就放在b的后面（前面）
    if( a.id > b.id )
    {
        return 1;
    }
    else
    {
        return -1;
    }
}

//也可以直接匿名函数
listShopItem.Sort(delegate( shopItem a, shopItem b )
{
    if( a.id > b.id )
    {
        return 1;
    }
    else
    {
        return -1;
    }
});

//甚至lambda表达式(配合三目运算符)
listShopItem.Sort((a, b) => { return a.id > b.id ? 1 : -1; });

题1

using System;
using System.Collections.Generic;

namespace Lesson16_List排序
{
    class Monster
    {
        public int atk;
        public int def;
        public int hp;

        public Monster(int atk, int def, int hp)
        {
            this.atk = atk;
            this.def = def;
            this.hp = hp;
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            List<Monster> monsters = new List<Monster>();
            Random r = new Random();
            for (int i = 0; i < 10; i++)
            {
                monsters.Add(new Monster(r.Next(20, 50), r.Next(10, 30), r.Next(100, 200)));
            }

            Console.WriteLine("输入1按攻击排序，输入2按防御排序，输入3按血量排序，输入4反转");
            try
            {
                int inputIndex = int.Parse(Console.ReadLine());

                switch (inputIndex)
                {
                    case 1:
                        monsters.Sort((a, b) => { return a.atk > b.atk ? 1 : -1; });
                        break;
                    case 2:
                        monsters.Sort((a, b) => { return a.def > b.def ? 1 : -1; });
                        break;
                    case 3:
                        monsters.Sort((a, b) => { return a.hp > b.hp ? 1 : -1; });
                        break;
                    case 4:
                        monsters.Reverse();
                        break;
                }
                for (int i = 0; i < monsters.Count; i++)
                {
                    Console.WriteLine("怪兽{0}的攻击力{1}，防御力{2}，血量{3}", i, monsters[i].atk, monsters[i].def, monsters[i].hp);
                }
            }
            catch
            {
                Console.WriteLine("请输入正确的数字");
            }
        }
    }
}

using System;
using System.Collections.Generic;

namespace Lesson16_List排序
{
    class Item
    {
        public int type;
        public string name;
        public int quality;

        public Item(int type, string name, int quality)
        {
            this.type = type;
            this.name = name;
            this.quality = quality;
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            List<Item> items = new List<Item>();
            Random r = new Random();
            Console.WriteLine("排序权重：类型 > 品质 > 名字长度");

            for (int i = 0; i < 10; i++)
            {
                items.Add(new Item(r.Next(1, 7), "Item" + r.Next(1, 300), r.Next(10, 100)));
                Console.WriteLine("物品类型{0}，名字{1}，品质{2}", items[i].type, items[i].name, items[i].quality);
            }

            items.Sort((a, b) =>
            {
                if (a.type != b.type)
                {
                    return a.type > b.type ? 1 : -1;
                }
                else if (a.quality != b.quality)
                {
                    return a.quality > b.quality ? -1 : 1;
                }
                else
                {
                    return (a.name).Length > (b.name).Length ? 1 : -1;
                }
            });

            Console.WriteLine("");

            for (int i = 0; i < 10; i++)
            {
                Console.WriteLine("物品类型{0}，名字{1}，品质{2}", items[i].type, items[i].name, items[i].quality);
            }
        }
    }
}

using System;
using System.Collections.Generic;

namespace Lesson16_List排序
{
    internal class Program
    {
        static void Main(string[] args)
        {
            Dictionary<int, int> dic = new Dictionary<int, int>();
            dic.Add(1, 88);
            dic.Add(2, 56);
            dic.Add(3, 66);
            dic.Add(4, 45);
            dic.Add(5, 23);
            dic.Add(6, 76);

            List<KeyValuePair<int, int>> list = new List<KeyValuePair<int, int>>();

            foreach (KeyValuePair<int, int> item in dic)
            {
                list.Add(item);
            }
            
            list.Sort(( a, b ) =>
            {
                return a.Value - b.Value;
            });

            for (int i = 0; i < list.Count; i++)
            {
                Console.WriteLine(list[i].Key + " " + list[i].Value);
            }
        }
    }
}
