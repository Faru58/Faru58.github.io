LinkedList

LinkedList

概念

List是一个C#为我们封装好的类，
它的本质是一个可变类型的泛型双向链表。

声明

using System.Collections.Generic;
    LinkedList<类型> linkedlist = new LinkedList<类型>();

两个类： 链表本身LinkedList 链表节点类LinkedListNode

增

//在链表尾部添加节点
linkedlist.AddLast(类型);
//在链表头部添加节点
linkedlist.AddFirst(类型);

//在某一个节点后添加节点
LinkedListNode<类型> node = LinkedList.Find(类型);
node.AddAfter(类型);
//在某一个节点前添加节点
LinkedListNode<类型> node = LinkedList.Find(类型);
node.AddBefore(类型);

删

//移除头节点
linkedlist.RemoveFirst();
//移除尾节点
linkedlist.RemoveLast();

//移除指定节点
linkedlist.Remove(类型);
//清空链表
linkedlist.Clear();

查

//查询头节点
LinkedListNode<类型> first = LinkedList.First;
//查询尾节点
LinkedListNode<类型> last = LinkedList.Last;

//找到指定值的节点（无法通过下标找，只能遍历查找）
LinkedListNode<类型> node = LinkedList.Find(类型);

//判断是否存在 bool
linkedlist.Contains(类型)

改

//直接改指定位置的元素
LinkedListNode<类型> node = LinkedList.Find(类型);
node.value = 类型;

遍历

//数组长度
list.Count
//数组容量
list.Capacity
//foreach遍历
foreach (类型 item in linkedlist)
{
    //直接取value
    Console.WriteLine(item);
}

//从头到尾遍历
LinkedListNode<类型> nowNode = LinkedList.First;
while( nowNode != null )
{
    Console.WriteLine(nowNode.value);
    nowNode = nowNode.Next;
}
//从尾到头遍历
LinkedListNode<类型> nowNode = LinkedList.Last;
while( nowNode != null )
{
    Console.WriteLine(nowNode.value);
    nowNode = nowNode.Previous;
}

练习题

using System;
using System.Collections.Generic;

namespace Lesson10_LinkedList
{
    internal class Program
    {
        static void Main(string[] args)
        {
            LinkedList<int> list = new LinkedList<int>();
            Random r = new Random();
            for (int i = 0; i < 10; i++)
            {
                list.AddLast(r.Next(100));
            }
            //正向遍历
            LinkedListNode<int> node = list.First;
            while (node != null)
            {
                Console.Write(node.Value + " ");
                node = node.Next;
            }
            Console.WriteLine("");
            //反向遍历
            node = list.Last;
            while (node != null)
            {
                Console.Write(node.Value + " ");
                node = node.Previous;
            }
        }
    }
}
