委托

概念

委托是函数（方法）的容器，可以理解为表示函数（方法）的变量类型，可以用来存储、传递函数（方法）
委托本质是一个类，用来定义函数（方法）的类型（返回值和参数的类型）
不同的函数（方法）必须对应各自“格式”一致的委托

声明

关键字： delegate

语法

访问修饰符 delegate 返回值 委托名(参数列表);
可以声明在namespace和class语句块中 更多的写在namespace中
函数声明语法前 + delegate

定义自定义委托

访问修饰符默认不写为public 在别的命名空间中也能使用
private 其他命名空间就不能使用了
一般使用public
同一个命名空间中委托不能重名

delegate void MyFun();
delegate int MyFun2(int value);

使用定义好的委托

委托变量是函数的容器

static void Fun()
{
    Console.WriteLine("");
}

MyFun f = new MyFun(Fun);
f.Invoke();

MyFun f2 = Fun;
f2();
static int Fun2(int value)
{
    return value;
}

MyFun2 f3 = new MyFun2(Fun2);
f3.Invoke();

MyFun2 f4 = Fun2;
f4(1);

委托常用在：
1.作为类的成员
2.作为函数的参数

class Test
{
    public MyFun fun;
    public MyFun2 fun2;

    public void TestFun( MyFun fun, MyFun2 fun2 )
    {
        //先处理一些别的逻辑 当这些逻辑处理完了 再执行传入的函数
        //
        //fun();
        //fun2(i);

        this.fun = fun;
        this.fun2 = fun2;
    }
}

Test t = new Test();
t.TestFun(Fun, Fun2);

委托变量可以存储多个函数（多播函数）

MyFun ff = Fun;
if( ff != null )
{
    ff();
}

增

public void AddFun( MyFun fun, MyFun2 fun2 )
{
    this.fun += fun;
    this.fun2 += fun2;
}

删

public void RemoveFun( MyFun fun, MyFun2 fun2 )
{
    this.fun -= fun;
    this.fun2 -= fun2;
}

系统定义好的委托

//无参无返回值
Action action = Fun;
action += Fun;
action();
//支持泛型的 可以让返回值和参数可变
Func<string> funcString = Fun4;
Func<int> funcInt = Fun5;

//可以传n个参数的 系统提供1-16个参数
Action<int, string, bool> action = Fun;
//可以传n个参数的 有返回值的
//第一个是in传入的, 第二个是out输出的
Func<int, int> func2 = Fun;

练习题

题1

using System;

namespace Lesson12_委托
{
    abstract class Person
    {
        public abstract void Eat();
    }

    class Mother : Person
    {
        public Action beginEat;

        public override void Eat()
        {
            Console.WriteLine("妈妈吃饭");
        }
        public void Cook()
        {
            Console.WriteLine("妈妈做饭");
            //执行委托函数
            if(beginEat != null)
            {
                beginEat();
            }
        }
    }

    class Father : Person
    {
        public override void Eat()
        {
            Console.WriteLine("爸爸吃饭");
        }
    }

    class Child : Person
    {
        public override void Eat()
        {
            Console.WriteLine("孩子吃饭");
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            Mother mother = new Mother();
            Father father = new Father();
            Child child = new Child();

            mother.beginEat += father.Eat;
            mother.beginEat += child.Eat;
            mother.beginEat += mother.Eat;

            mother.Cook();
        }
    }
}

题2

using System;

namespace Lesson12_委托
{
    class Monster
    {
        public Action<Monster> die;
        public int money = 10;

        public void Dead()
        {
            if (die != null)
            {
                die(this);
            }
            //一般情况下 委托关联的函数 有加就有减 或直接清空
            die = null;
        }
    }

    class Player
    {
        private int nowMoney = 0;
        public void MonsterDie( Monster monster )
        {
            nowMoney += monster.money;
            Console.WriteLine("当前面板显示{0}钱", nowMoney);
        }
    }

    class Panel
    {
        private int nowMoney = 0;
        public void MonsterDie( Monster monster )
        {
            nowMoney += monster.money;
            Console.WriteLine("当前面板显示{0}钱", nowMoney);
        }
    }

    class Achievement
    {
        private int nowKill= 0;
        public void MonsterDie(Monster monster)
        {
            nowKill++;
            Console.WriteLine("当前击杀{0}怪物", nowKill);
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            Monster monster = new Monster();
            Player player = new Player();
            Panel panel = new Panel();
            Achievement achievement = new Achievement();

            monster.die += player.MonsterDie;
            monster.die += panel.MonsterDie;
            monster.die += achievement.MonsterDie;

            monster.Dead();
        }
    }
}
