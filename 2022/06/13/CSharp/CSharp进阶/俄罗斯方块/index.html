using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    internal class BlockInfo
    {
        //方块信息坐标的容器,必须要初始化
        private List<Position[]> blockPosList = new List<Position[]>(); 

        public BlockInfo(E_DrawType type)
        {
            switch (type)
            {
                case E_DrawType.Cube:
                    blockPosList.Add(new Position[3] { 
                        new Position(2, 0),
                        new Position(0, 1),
                        new Position(2, 1) 
                    });
                    break;
                case E_DrawType.Line:
                    //初始化 长条形状的4种形态的坐标信息
                    blockPosList.Add(new Position[3] {
                        new Position(0, -1),
                        new Position(0, 1),
                        new Position(0, 2)
                    }); //blockPosList索引[0]
                    blockPosList.Add(new Position[3] {
                        new Position(-4, 0),
                        new Position(-2, 0),
                        new Position(2, 0)
                    }); //blockPosList索引[1]
                    blockPosList.Add(new Position[3] {
                        new Position(0, -2),
                        new Position(0, -1),
                        new Position(0, 1)
                    }); //blockPosList索引[2]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 0),
                        new Position(2, 0),
                        new Position(4, 0)
                    }); //blockPosList索引[3]
                    break;
                case E_DrawType.Tank:
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 0),
                        new Position(0, 1),
                        new Position(2, 0)
                    }); //blockPosList索引[0]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 0),
                        new Position(0, -1),
                        new Position(0, 1)
                    }); //blockPosList索引[1]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 0),
                        new Position(0, -1),
                        new Position(2, 0)
                    }); //blockPosList索引[2]
                    blockPosList.Add(new Position[3] {
                        new Position(0, -1),
                        new Position(0, 1),
                        new Position(2, 0)
                    }); //blockPosList索引[3]
                    break;
                case E_DrawType.Left_Ladder:
                    blockPosList.Add(new Position[3] {
                        new Position(0, -1),
                        new Position(2, 1),
                        new Position(2, 0)
                    }); //blockPosList索引[0]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 1),
                        new Position(2, 0),
                        new Position(0, 1)
                    }); //blockPosList索引[1]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, -1),
                        new Position(-2, 0),
                        new Position(0, 1)
                    }); //blockPosList索引[2]
                    blockPosList.Add(new Position[3] {
                        new Position(0, -1),
                        new Position(2, -1),
                        new Position(-2, 0)
                    }); //blockPosList索引[3]
                    break;
                case E_DrawType.Right_Ladder:
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 1),
                        new Position(0, -1),
                        new Position(-2, 0)
                    }); //blockPosList索引[0]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, -1),
                        new Position(0, -1),
                        new Position(2, 0)
                    }); //blockPosList索引[1]
                    blockPosList.Add(new Position[3] {
                        new Position(2, 0),
                        new Position(0, 1),
                        new Position(2, -1)
                    }); //blockPosList索引[2]
                    blockPosList.Add(new Position[3] {
                        new Position(2, 1),
                        new Position(0, 1),
                        new Position(-2, 0)
                    }); //blockPosList索引[3]
                    break;
                case E_DrawType.Left_Long_Ladder:
                    blockPosList.Add(new Position[3] {
                        new Position(-2, -1),
                        new Position(0, 1),
                        new Position(0, -1)
                    }); //blockPosList索引[0]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 0),
                        new Position(2, 0),
                        new Position(2, -1)
                    }); //blockPosList索引[1]
                    blockPosList.Add(new Position[3] {
                        new Position(0, 1),
                        new Position(0, -1),
                        new Position(2, 1)
                    }); //blockPosList索引[2]
                    blockPosList.Add(new Position[3] {
                        new Position(2, 0),
                        new Position(-2, 1),
                        new Position(-2, 0)
                    }); //blockPosList索引[3]
                    break;
                case E_DrawType.Right_Long_Ladder:
                    blockPosList.Add(new Position[3] {
                        new Position(2, -1),
                        new Position(0, -1),
                        new Position(0, 1)
                    }); //blockPosList索引[0]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 0),
                        new Position(2, 0),
                        new Position(2, 1)
                    }); //blockPosList索引[1]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, 1),
                        new Position(0, -1),
                        new Position(0, 1)
                    }); //blockPosList索引[2]
                    blockPosList.Add(new Position[3] {
                        new Position(-2, -1),
                        new Position(-2, 0),
                        new Position(2, 0)
                    }); //blockPosList索引[3]
                    break;
            }
        }

        /// <summary>
        /// 提供给外部根据索引快速获取位置偏移信息的
        /// </summary>
        /// <param name="index">第几种形态的索引</param>
        /// <returns></returns>
        public Position[] this[int index]
        {
            get
            {
                if(index < 0)
                    return blockPosList[0];
                else if(index >= blockPosList.Count)
                    return blockPosList[blockPosList.Count - 1];
                else
                    return blockPosList[index];
            }
        }

        /// <summary>
        /// 提供给外部获取形态有几种
        /// </summary>
        public int Count { get => blockPosList.Count; }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    /// <summary>
    /// 变形方向枚举
    /// </summary>
    enum E_ChangeType
    {
        /// <summary>
        /// 逆时针
        /// </summary>
        Left,
        /// <summary>
        /// 顺时针
        /// </summary>
        Right,
    }
    internal class Carrier
    {
        private List<DrawObject> blocks;
        private Dictionary<E_DrawType, BlockInfo> dic;
        private BlockInfo nowBlockInfo;
        private int nowInfoIndex;

        public Carrier()
        {
            dic = new Dictionary<E_DrawType, BlockInfo>()
            {
                { E_DrawType.Cube, new BlockInfo(E_DrawType.Cube) },
                { E_DrawType.Line, new BlockInfo(E_DrawType.Line) },
                { E_DrawType.Tank, new BlockInfo(E_DrawType.Tank) },
                { E_DrawType.Left_Ladder, new BlockInfo(E_DrawType.Left_Ladder) },
                { E_DrawType.Right_Ladder, new BlockInfo(E_DrawType.Right_Ladder) },
                { E_DrawType.Left_Long_Ladder, new BlockInfo(E_DrawType.Left_Long_Ladder) },
                { E_DrawType.Right_Long_Ladder, new BlockInfo(E_DrawType.Right_Long_Ladder) },
            };

            RandomCreateBlock();
        }
            
        /// <summary>
        /// 随机创建一个方块
        /// </summary>
        public void RandomCreateBlock()
        {
            Random r = new Random();
            E_DrawType t = (E_DrawType)r.Next(1, 8);
            //新建方块 = 创建4个小方块
            blocks = new List<DrawObject>()
            {
                new DrawObject(t),
                new DrawObject(t),
                new DrawObject(t),
                new DrawObject(t),
            };
            //初始化原点方块，每次都从中间出来
            blocks[0].pos = new Position(24, -5);
            //取出方块形态信息
            nowBlockInfo = dic[t];
            //随机确定现在是哪种形态,之后旋转改变
            nowInfoIndex = r.Next(0, dic.Count);
            Position[] pos = nowBlockInfo[nowInfoIndex];
            for (int i = 0; i < pos.Length; i++)
            {
                //取出来的坐标是相对原点方块的坐标
                blocks[i + 1].pos = blocks[0].pos + pos[i];
            }
        }

        public void Draw()
        {
            for (int i = 0; i < blocks.Count; i++)
            {
                blocks[i].Draw();
            }
        }

        public void CleanDraw()
        {
            for (int i = 0; i < blocks.Count; i++)
            {
                blocks[i].CleanDraw();
            }
        }

        /// <summary>
        /// 变形旋转的方法
        /// </summary>
        /// <param name="type">旋转的方向</param>
        public void Change(E_ChangeType type)
        {
            CleanDraw();
            switch(type)
            {
                case E_ChangeType.Left:
                    --nowInfoIndex;
                    if (nowInfoIndex < 0)
                        nowInfoIndex = nowBlockInfo.Count - 1;
                    break;
                case E_ChangeType.Right:
                    ++nowInfoIndex;
                    if(nowInfoIndex >= nowBlockInfo.Count)
                        nowInfoIndex = 0;
                    break;
            }
            //设置剩下三个小方块位置
            Position[] pos = nowBlockInfo[nowInfoIndex];
            for (int i = 0; i < pos.Length; i++)
            {
                //取出来的坐标是相对原点方块的坐标
                blocks[i + 1].pos = blocks[0].pos + pos[i];
            }
            Draw();
        }

        /// <summary>
        /// 判断是否可以进行变形
        /// </summary>
        /// <param name="type">变形方向</param>
        /// <param name="map">地图信息</param>
        /// <returns></returns>
        public bool CanChange(E_ChangeType type, Map map)
        {
            //用一个临时变量记录当前索引，不变化当前索引
            int nowIndex = nowInfoIndex;
            switch (type)
            {
                case E_ChangeType.Left:
                    --nowIndex;
                    if (nowIndex < 0)
                        nowIndex = nowBlockInfo.Count - 1;
                    break;
                case E_ChangeType.Right:
                    ++nowIndex;
                    if (nowIndex >= nowBlockInfo.Count)
                        nowIndex = 0;
                    break;
            }
            Position[] nowPos = nowBlockInfo[nowIndex];
            //判断是否超出地图边界
            Position tempPos;
            for (int i = 0; i < nowPos.Length; i++)
            {
                tempPos = blocks[0].pos + nowPos[i];
                if( tempPos.x < 2 ||
                    tempPos.x >= Game.w - 2 ||
                    tempPos.y >= Game.h - 6 )
                {
                    return false;
                }
            }
            //判断是否和地图上的动态方块重合
            for (int i = 0; i < nowPos.Length; i++)
            {
                tempPos = blocks[0].pos + nowPos[i];
                for (int j = 0; j < map.dynamicWalls.Count; j++)
                {
                    if(tempPos == map.dynamicWalls[j].pos)
                    {
                        return false;
                    }
                }
            }
            return true;
        }

        /// <summary>
        /// 左右移动的函数
        /// </summary>
        /// <param name="type">左或者右</param>
        public void MoveRL(E_ChangeType type)
        {
            CleanDraw();
            Position movePos = new Position(type == E_ChangeType.Left ? -2 : 2, 0);
            for (int i = 0; i < blocks.Count; i++)
            {
                blocks[i].pos += movePos;
            }          
            Draw();
        }

        /// <summary>
        /// 移动之前判断是否可以左右移动
        /// </summary>
        /// <param name="type">移动的方向</param>
        /// <param name="map"></param>
        public bool CanMoveRl(E_ChangeType type, Map map)
        {
            Position movePos = new Position(type == E_ChangeType.Left ? -2 : 2, 0);
            //动过后的结果不能直接改变，用一个临时变量存着
            Position pos;
            for (int i = 0; i < blocks.Count; i++)
            {
                pos = blocks[i].pos + movePos;
                if( pos.x < 2 || pos.x >= Game.w - 2 )
                    return false;
            }
            for (int i = 0; i < blocks.Count; i++)
            {
                pos = blocks[i].pos + movePos;
                for (int j = 0; j < map.dynamicWalls.Count; j++)
                {
                    if (pos == map.dynamicWalls[j].pos)
                    {
                        return false;
                    }
                }
            }
            return true;
        }

        /// <summary>
        /// 自动向下移动
        /// </summary>
        public void AutoMove()
        {
            CleanDraw();
            Position downMove = new Position(0, 1);
            for (int i = 0; i < blocks.Count; i++)
            {
                blocks[i].pos += downMove;
            }
            Draw();
        }

        /// <summary>
        /// 判断是否停下自动向下移动
        /// </summary>
        /// <param name="map"></param>
        /// <returns></returns>
        public bool CanAutoMove(Map map)
        {
            Position movePos = new Position(0, 1);
            //动过后的结果不能直接改变，用一个临时变量存着
            Position pos;
            for (int i = 0; i < blocks.Count; i++)
            {
                pos = blocks[i].pos + movePos;
                if (pos.y >= Game.h - 6)
                {
                    map.AddWalls(blocks);
                    RandomCreateBlock();
                    return false;
                }               
            }
            for (int i = 0; i < blocks.Count; i++)
            {
                pos = blocks[i].pos + movePos;
                for (int j = 0; j < map.dynamicWalls.Count; j++)
                {
                    if (pos == map.dynamicWalls[j].pos)
                    {
                        map.AddWalls(blocks);
                        RandomCreateBlock();
                        return false;
                    }
                }
            }
            return true;
        }

    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    /// <summary>
    /// 绘制类型，根据不同类型改变绘制方块的颜色
    /// </summary>
    enum E_DrawType
    {
        /// <summary>
        /// 墙壁
        /// </summary>
        Wall,
        /// <summary>
        /// 正方形
        /// </summary>
        Cube,
        /// <summary>
        /// 直线
        /// </summary>
        Line,
        /// <summary>
        /// 坦克
        /// </summary>
        Tank,
        /// <summary>
        /// 左梯子
        /// </summary>
        Left_Ladder,
        /// <summary>
        /// 右梯子
        /// </summary>
        Right_Ladder,
        /// <summary>
        /// 左长梯子
        /// </summary>
        Left_Long_Ladder,
        /// <summary>
        /// 右长梯子
        /// </summary>
        Right_Long_Ladder,
    }
    class DrawObject : IDraw
    {
        public Position pos;
        public E_DrawType type;

        public DrawObject(E_DrawType type)
        {
            this.type = type;
        }

        public DrawObject(E_DrawType type, int x, int y) : this(type)
        {
            this.pos = new Position(x, y);
        }

        public void CleanDraw()
        {
            //若在屏幕外不需要绘制
            if (pos.y < 0)
                return;
            Console.SetCursorPosition(pos.x, pos.y);
            Console.Write("  ");
        }

        public void Draw()
        {
            //若在屏幕外不需要绘制
            if (pos.y < 0)
                return;
            Console.SetCursorPosition(pos.x, pos.y);
            //设置颜色
            switch (type)
            {
                case E_DrawType.Wall:
                    Console.ForegroundColor = ConsoleColor.Red;
                    break;
                case E_DrawType.Cube:
                    Console.ForegroundColor = ConsoleColor.Blue;
                    break;
                case E_DrawType.Line:
                    Console.ForegroundColor = ConsoleColor.Green;
                    break;
                case E_DrawType.Tank:
                    Console.ForegroundColor = ConsoleColor.Cyan;
                    break;
                case E_DrawType.Left_Ladder:
                case E_DrawType.Right_Ladder:
                    Console.ForegroundColor = ConsoleColor.Magenta;
                    break;
                case E_DrawType.Left_Long_Ladder:
                case E_DrawType.Right_Long_Ladder:
                    Console.ForegroundColor = ConsoleColor.DarkYellow;
                    break;
            }
            Console.Write("■");
        }

        /// <summary>
        /// 切换方块类型（变成墙壁）
        /// </summary>
        /// <param name="type">想要变成的方块类型</param>
        public void ChangeType(E_DrawType type)
        {
            this.type = type;
        }

    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    struct Position
    {
        public int x;
        public int y;

        public Position(int x, int y)
        {
            this.x = x; this.y = y;
        }

        public static bool operator == (Position p1, Position p2)
        {
            if(p1.x == p2.x && p1.y == p2.y)
                return true;
            return false;
        }

        public static bool operator !=(Position p1, Position p2)
        {
            if (p1.x == p2.x && p1.y == p2.y)
                return false;
            return true;
        }
        public static Position operator +(Position p1, Position p2)
        {
            Position p = new Position(p1.x + p2.x, p1.y + p2.y);
            return p;
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    internal class Map : IDraw
    {
        //固定墙壁
        private List<DrawObject> fixedWalls = new List<DrawObject>();
        //动态墙壁
        public List<DrawObject> dynamicWalls = new List<DrawObject>();
        //记录每一层有多少小方块的容器，索引就是第几行
        private int[] recordInfo;
        public int w = 0;
        public int h;
        public int score = 0;
        private GameScene nowGameScene;

        //重载一次无参构造去初始化固定墙壁
        public Map( GameScene nowGameScene )
        {
            this.nowGameScene = nowGameScene;
            h = Game.h - 6;
            //重点：记录行号的容器
            recordInfo = new int[h];
            //底下墙壁
            for (int i = 0; i < Game.w; i += 2)
            {
                fixedWalls.Add(new DrawObject(E_DrawType.Wall, i, Game.h - 6));
                w++;
            }
            
            //两边墙壁
            for (int i = 0; i < Game.h - 6; i++)
            {
                fixedWalls.Add(new DrawObject(E_DrawType.Wall, 0, i));
                fixedWalls.Add(new DrawObject(E_DrawType.Wall, Game.w - 2, i));
            }


        }

        /// <summary>
        /// 绘制墙壁
        /// </summary>
        public void Draw()
        {
            //画静态墙壁
            for (int i = 0; i < fixedWalls.Count; i++)
            {
                fixedWalls[i].Draw();
            }
            //画动态墙壁
            for (int i = 0; i < dynamicWalls.Count; i++)
            {
                dynamicWalls[i].Draw();
            }
        }

        /// <summary>
        /// 擦除动态墙壁，以便下移
        /// </summary>
        public void CleanDynamicWalls()
        {
            for (int i = 0; i < dynamicWalls.Count; i++)
            {
                dynamicWalls[i].CleanDraw();
            }
        }

        /// <summary>
        /// 提供给外部添加动态方块的函数
        /// </summary>
        /// <param name="walls"></param>
        public void AddWalls(List<DrawObject> walls)
        {
            for (int i = 0; i < walls.Count; i++)
            {
                walls[i].ChangeType(E_DrawType.Wall);
                dynamicWalls.Add(walls[i]);
                //在动态墙壁添加时，若出地图，则停止
                if(walls[i].pos.y <= 0)
                {
                    //关闭输入线程
                    nowGameScene.StopTheard();
                    Game.ChangeScene(E_SceneType.End);
                    break;
                }
                recordInfo[Game.h - 7 - walls[i].pos.y]++;
            }
            CleanDynamicWalls();
            CheckClear();
            Draw();
        }

        /// <summary>
        /// 检测是否消层
        /// </summary>
        public void CheckClear()
        {
            List<DrawObject> delList = new List<DrawObject>();
            for (int i = 0; i < recordInfo.Length; i++)
            {
                if(recordInfo[i] == w - 2)
                {
                    for (int j = 0; j < dynamicWalls.Count; j++)
                    {
                        if( i == Game.h - 7 - dynamicWalls[j].pos.y )
                        {
                            //为了安全移除，添加一个待移除列表
                            delList.Add(dynamicWalls[j]);
                        }
                        else if( i < Game.h - 7 - dynamicWalls[j].pos.y )
                        {
                            //把该行上面的方块下移，即纵坐标+1
                            dynamicWalls[j].pos.y++;
                        }
                    }
                    //移除待移除列表里的方块
                    for (int j = 0; j < delList.Count; j++)
                    {
                        dynamicWalls.Remove(delList[j]);                       
                    }
                    //记录小方块数量的数组从上到下迁移
                    for (int j = i; j < recordInfo.Length - 1; j++)
                    {
                        recordInfo[j] = recordInfo[j + 1];
                    }
                    score++;
                    //通过递归，实现一次移除多行
                    recordInfo[recordInfo.Length - 1] = 0;
                    CheckClear();
                    break;
                }
            }
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    internal class InputThread
    {
        //线程成员变量
        Thread inputThread;
        //输入检测事件
        public event Action inputEvent;

        //单例模式
        private static InputThread instance = new InputThread();

        public static InputThread Instance
        {
            get { return instance; }
        }

        private InputThread()
        {
            inputThread = new Thread(InputCheck);
            inputThread.IsBackground = true;
            inputThread.Start();
        }

        private void InputCheck()
        {
            while (true)
            {
                inputEvent?.Invoke();
            }
        }

    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    //抽象基类
    abstract class BeginOrEndBaseScene:ISceneUpdate
    {
        protected int nowSelIndex;
        protected string strTitle;
        protected string strOne;

        public abstract void EnterJDoSomething();

        public void Update()
        {
            Console.ForegroundColor = ConsoleColor.White;
            //显示标题
            Console.SetCursorPosition( Game.w / 2 - strTitle.Length, 5);
            Console.WriteLine(strTitle);


            //显示下方选项
            Console.SetCursorPosition(Game.w / 2 - strOne.Length, 8);
            Console.ForegroundColor = nowSelIndex == 0 ? ConsoleColor.Red : ConsoleColor.White;
            Console.WriteLine(strOne);
            Console.SetCursorPosition(Game.w / 2 - 4, 10);
            Console.ForegroundColor = nowSelIndex == 1 ? ConsoleColor.Red : ConsoleColor.White;
            Console.WriteLine("结束游戏");

            //检测输入
            switch(Console.ReadKey(true).Key)
            {
                case ConsoleKey.W:
                    --nowSelIndex;
                    if(nowSelIndex < 0)
                    {
                        nowSelIndex = 0;
                    }
                    break;
                case ConsoleKey.S:
                    ++nowSelIndex;
                    if (nowSelIndex > 1)
                    {
                        nowSelIndex = 1;
                    }
                    break;
                case ConsoleKey.J:
                    EnterJDoSomething();
                    break;
            }

        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    /// <summary>
    /// 场景类型枚举
    /// </summary>
    enum E_SceneType
    {
        /// <summary>
        /// 开始场景
        /// </summary>
        Begin,
        /// <summary>
        /// 游戏场景
        /// </summary>
        Game,
        /// <summary>
        /// 结束场景
        /// </summary>
        End,
    }

    class Game
    {
        //游戏窗口宽高
        public const int w = 50;
        public const int h = 35;
        //当前选中的场景
        public static ISceneUpdate nowScene;

        //初始化的方法
        public Game()
        {
            Console.CursorVisible = false;
            Console.SetWindowSize(w, h);
            Console.SetBufferSize(w, h);
            ChangeScene(E_SceneType.Begin);
        }

        //游戏开始的方法
        public void Start()
        {
            //游戏主循环 负责游戏场景的更新
            while(true)
            {
                //判断当前游戏场景不为空 就更新
                if(nowScene != null)
                {
                    nowScene.Update();
                }
            }
        }

        //切换场景的方法
        public static void ChangeScene(E_SceneType type)
        {
            //切场景之前，应该清空上一个场景的内容
            Console.Clear();

            switch (type)
            {
                case E_SceneType.Begin:
                    nowScene = new BeginScene();
                    break;
                case E_SceneType.Game:
                    nowScene = new GameScene();
                    break;
                case E_SceneType.End:
                    nowScene = new EndScene();
                    break;
            }
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

namespace 俄罗斯方块
{
    class GameScene: ISceneUpdate
    {
        Map map;
        Carrier carrier;
        //Thread inputThread;
        //bool isRunning = true;
        public GameScene()
        {           
            map = new Map(this);
            carrier = new Carrier();

            //添加输入事件监听
            InputThread.Instance.inputEvent += CheckInputThread;
            //inputThread = new Thread(CheckInputThread);
            //inputThread.IsBackground = true;
            //inputThread.Start();
        }

        /// <summary>
        /// 停止线程
        /// </summary>
        public void StopTheard()
        {
            //isRunning = false;
            //inputThread = null;

            //移除输入事件监听
            InputThread.Instance.inputEvent -= CheckInputThread;
        }

        private void CheckInputThread()
        {
            while (true)
            {
                //在控制台中 检测玩家输入 让程序不被检测卡主
                //判断 有没有键盘输入 如果有 才为true
                if (Console.KeyAvailable)
                {
                    //为避免影响主线程 在输入后加锁
                    lock(carrier)
                    {
                        //检测输入输出 不能再 间隔帧里面去处理 应该每次都检测 这样才准确
                        switch (Console.ReadKey(true).Key)
                        {
                            case ConsoleKey.S:
                                if (carrier.CanAutoMove(map))
                                    carrier.AutoMove();
                                break;
                            case ConsoleKey.A:
                                if (carrier.CanMoveRl(E_ChangeType.Left, map))
                                    carrier.MoveRL(E_ChangeType.Left);
                                break;
                            case ConsoleKey.D:
                                if (carrier.CanMoveRl(E_ChangeType.Right, map))
                                    carrier.MoveRL(E_ChangeType.Right);
                                break;
                            case ConsoleKey.LeftArrow:
                                if (carrier.CanChange(E_ChangeType.Left, map))
                                    carrier.Change(E_ChangeType.Left);
                                break;
                            case ConsoleKey.RightArrow:
                                if (carrier.CanChange(E_ChangeType.Right, map))
                                    carrier.Change(E_ChangeType.Right);
                                break;
                        }
                    }                   
                }
            }
        }

        public void Update()
        {
            lock(carrier)
            {
                map.Draw();
                carrier.Draw();
                if (carrier.CanAutoMove(map))
                    carrier.AutoMove();
                Console.SetCursorPosition(2, Game.h - 4);
                Console.ForegroundColor = ConsoleColor.White;
                Console.WriteLine("当前得分：{0}", map.score);
            }          
            Thread.Sleep(350);                  
        }
    }
}
