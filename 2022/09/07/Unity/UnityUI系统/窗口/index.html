窗口

第一个参数：id 窗口唯一的ID，不能合其他窗口重复
第三个参数：委托函数参数 用于绘制窗口的函数，传入即可

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Lesson8 : MonoBehaviour
{
    private void OnGUI()
    {
        //窗口
        GUI.Window(1, new Rect(100, 100, 100, 100), DrawWindow, "测试窗口1");
        //id除了区分不同窗口，还可以在一个函数中去处理多个窗口的逻辑
        GUI.Window(2, new Rect(300, 100, 100, 100), DrawWindow, "测试窗口2");
    }


    private void DrawWindow(int id)
    {
        switch (id)
        {
            case 1:
                GUI.Button(new Rect(0, 30, 30, 20), "1");
                break;
            case 2:
                GUI.Button(new Rect(0, 30, 30, 20), "2");
                break;
        }
    }
}

模态窗口

模态窗口能让其他控件失效，只能先处理该窗口（点击该窗口上控件），才能处理其他窗口逻辑

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Lesson8 : MonoBehaviour
{
    private void OnGUI()
    {
        //模态窗口
        GUI.ModalWindow(3, new Rect(500, 100, 100, 100), DrawWindow, "模态窗口3");
    }


    private void DrawWindow(int id)
    {
        switch (id)
        {
            case 3:
                GUI.Button(new Rect(0, 30, 30, 20), "3");
                break;
        }
    }
}

拖动窗口

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Lesson8 : MonoBehaviour
{
    private Rect dragWindowPos = new Rect(400, 400, 200, 150);

    private void OnGUI()
    {
        //拖动窗口
        //位置赋值只是前提
        dragWindowPos = GUI.Window(4, dragWindowPos, DrawWindow, "拖动窗口");
    }


    private void DrawWindow(int id)
    {
        switch (id)
        {
                //该API写在委托中调用，才能让窗口被拖动
                //传入Rect参数的重载，决定窗口中哪一部分位置可以被拖动（不填默认全部）
                GUI.DragWindow();
                break;
        }
    }
}

QuitTipPanel.cs

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class QuitTipPanel : MonoBehaviour
{
    //为使面板的显示隐藏 所有地方都能快速使用
    //静态方法和静态变量 可以直接通过类名调用
    private static QuitTipPanel instance;

    public static void ShowMe()
    {
        if (instance != null)
        {
            instance.gameObject.SetActive(true);
        }
    }
    public static void HideMe()
    {
        if (instance != null)
        {
            instance.gameObject.SetActive(false);
        }
    }

    public Rect quitWindowPos;
    public Rect labelPos;
    public Rect button1Pos;
    public Rect button2Pos;

    private void Awake()
    {
        instance = this;
        HideMe();
    }


    private void OnGUI()
    {
        GUI.ModalWindow(1, quitWindowPos, DrawWindow, "提示");
    }

    private void DrawWindow(int id)
    {
        GUI.Label(labelPos, "是否退出游戏");
        if (GUI.Button(button1Pos, "退出"))
        {
            Application.Quit();
        }
        if (GUI.Button(button2Pos, "取消"))
        {
            HideMe();
            BeginPanel.ShowMe();
        }
    }
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.SceneManagement;

public class BeginPanel : MonoBehaviour
{
    //为使面板的显示隐藏 所有地方都能快速使用
    //静态方法和静态变量 可以直接通过类名调用
    private static BeginPanel instance;

    public static void ShowMe()
    {
        if(instance != null)
        {
            instance.gameObject.SetActive(true);
        }
    }
    public static void HideMe()
    {
        if (instance != null)
        {
            instance.gameObject.SetActive(false);
        }
    }

    public Rect labelPos;
    public GUIContent labelContent;
    public GUIStyle labelStyle;

    public Rect button1Pos;
    public Rect button2Pos;
    public Rect button3Pos;

    public GUIStyle buttonStyle;

    public Rect bkPos;
    public Texture bkImage;
    public ScaleMode bkMode = ScaleMode.ScaleToFit;

    private void Awake()
    {
        instance = this;
    }

    private void OnGUI()
    {
        //设置背景
        GUI.DrawTexture(bkPos, bkImage, bkMode);

        //游戏标题
        GUI.Label(labelPos, labelContent, labelStyle);

        //3个游戏按钮
        if(GUI.Button(button1Pos, "开始游戏", buttonStyle))
        {
            //SceneManager.LoadScene("GameScene");
            HideMe();
            LoginPanel.ShowMe();
        }
        if(GUI.Button(button2Pos, "设置游戏", buttonStyle))
        {
            HideMe();
            SettingPanel.ShowMe();
        }
        if(GUI.Button(button3Pos, "结束游戏", buttonStyle))
        {
            HideMe();
            QuitTipPanel.ShowMe();
        }


    }

    
}
