数据类型存储

using System;
using System.Collections;
using System.Collections.Generic;
using System.Reflection;
using UnityEngine;

/// <summary>
/// PlayerPrefs数据管理类 统一管理数据的存储和读取
/// </summary>
public class PlayerPrefsDataMgr
{
    private static PlayerPrefsDataMgr instance = new PlayerPrefsDataMgr();

    public static PlayerPrefsDataMgr Instance
    {
        get { return instance; }
    }

    private PlayerPrefsDataMgr()
    {

    }

    /// <summary>
    /// 存储数据
    /// </summary>
    /// <param name="data">数据对象</param>
    /// <param name="keyName">数据对象的唯一key 自己控制</param>
    public void SaveData(object data, string keyName)
    {
        Type dataType = data.GetType();
        //1.得到传入数据对象所有的字段
        FieldInfo[] infos = dataType.GetFields();
        for (int i = 0; i < infos.Length; i++)
        {
            Debug.Log(infos[i]);
        }

        //2.自定义一个key的规则进行数据存储(保证key的唯一性)
        //keyName_数据类型_字段类型_字段名
        string saveKeyName = "";
        FieldInfo field;
        for (int i = 0; i < infos.Length; i++)
        {
            field = infos[i];
            //对每一个字段进行存储
            saveKeyName = keyName + "_" + dataType.Name + "_" + field.FieldType.Name + "_" + field.Name;
            //封装一个方法专门存储
            SaveValue(field.GetValue(data), saveKeyName);
        }
    }

    private void SaveValue(object value, string keyName)
    {
        //常用数据类型存储
        //PlayerPrefs只支持3种类型存储，故需要判断value类型
        Type fieldType = value.GetType();
        if (fieldType == typeof(int))
        {
            PlayerPrefs.SetInt(keyName, (int)value);
        }
        else if (fieldType == typeof(float))
        {
            PlayerPrefs.SetFloat(keyName, (float)value);
        }
        else if (fieldType == typeof(string))
        {
            PlayerPrefs.SetString(keyName, value.ToString());
        }
        else if (fieldType == typeof(bool))
        {
            PlayerPrefs.SetInt(keyName, (bool)value ? 1 : 0);
        }
        //List数据类型存储
        //通过反射判断父子关系
        //判断字段类型是不是IList的子类(因为不确定List的泛型)
        else if (typeof(IList).IsAssignableFrom(fieldType))
        {
            Debug.Log("存储List " + keyName);
            //父类装子类
            IList list = value as IList;
            //先存储数量
            PlayerPrefs.SetInt(keyName, list.Count);
            int index = 0;
            foreach (object item in list)
            {
                //存储具体的值
                SaveValue(item, keyName + index);
                ++index;
            }
        }
        //Dictionary数据类型存储
        //判断字段类型是不是IDictionary的子类(因为不确定Dictionary的泛型)
        else if (typeof(IDictionary).IsAssignableFrom(fieldType))
        {
            Debug.Log("存储Dictionary " + keyName);
            //父类装子类
            IDictionary dictionary = value as IDictionary;
            //先存储数量
            PlayerPrefs.SetInt(keyName, dictionary.Count);
            int index = 0;
            foreach (object key in dictionary.Keys)
            {
                //存储具体的值
                SaveValue(key, keyName + "_key_" + index);
                SaveValue(dictionary[key], keyName + "_value_" + index);
                ++index;
            }
        }
        //自定义类成员存储
        //基础数据类型都不是，可能就是自定义类型
        else
        {
            SaveData(value, keyName);
        }
    }
}

读取数据类型

using System;
using System.Collections;
using System.Collections.Generic;
using System.Reflection;
using UnityEngine;

/// <summary>
/// PlayerPrefs数据管理类 统一管理数据的存储和读取
/// </summary>
public class PlayerPrefsDataMgr
{
    private static PlayerPrefsDataMgr instance = new PlayerPrefsDataMgr();

    public static PlayerPrefsDataMgr Instance
    {
        get { return instance; }
    }

    private PlayerPrefsDataMgr()
    {

    }

    /// <summary>
    /// 读取数据
    /// </summary>
    /// <param name="type">想要读取的数据的数据类型</param>
    /// <param name="keyName">数据对象的唯一key 自己控制</param>
    /// <returns></returns>
    public object LoadData(Type type, string keyName)
    {
        //不用object对象传入，而使用Type传入
        //主要目的是节约一行代码(在外部)
        //假设现在要读取一个Player类型的数据，如果是object，就必须在外部new一个object对象传入
        //现在用Type只需传入一个Type typeof(Player)，就可以在内部动态创建一个对象返回出去
        //达到了在外部少写一行代码的作用

        //根据传入的类型和keyName，依据存储数据时key的拼接规则，来进行数据的获取赋值并返回

        //根据传入的Type创建一个对象，用于存储数据
        //Activator.CreateInstance 使用与指定参数匹配程度最高的构造函数来创建指定类型的实例
        //创建一个方法 方法中传入一个类的名称 然后就能返回一个这个类的实例 ，这样的做法让程序有更高的拓展性
        object data = Activator.CreateInstance(type); 
        //往new出来的对象中存储数据
        //得到所有字段
        FieldInfo[] infos = type.GetFields();
        //用于拼接key的字符串
        string loadKeyName = "";
        //用于存储单个字段信息的对象
        FieldInfo info;
        for (int i = 0; i < infos.Length; i++)
        {
            info = infos[i];
            //key的拼接规则一定和存储引用，不然找不到对应数据
            loadKeyName = keyName + "_" + type.Name + "_" + info.FieldType.Name + "_" + info.Name;
            //有key就可以结合PlayerPrefs来读取数据
            //填充数据到data中
            info.SetValue(data, LoadValue(info.FieldType, loadKeyName));
        }
        return data;
    }

    private object LoadValue(Type fieldType, string keyName)
    {
        //根据字段判断用哪个API来读取
        //读取常用数据类型
        if( fieldType == typeof(int) )
        {
            Debug.Log("读取int " + keyName);
            return PlayerPrefs.GetInt(keyName, 0);
        }
        else if( fieldType == typeof(float) )
        {
            return PlayerPrefs.GetFloat(keyName, 0);
        }
        else if( fieldType == typeof(string) )
        {
            return PlayerPrefs.GetString(keyName, "");
        }
        else if( fieldType == typeof(bool) )
        {
            return PlayerPrefs.GetInt(keyName, 0) == 1 ? true : false;
        }
        //读取List数据类型
        else if( typeof(IList).IsAssignableFrom(fieldType) )
        {
            //得到长度
            int count = PlayerPrefs.GetInt(keyName, 0);
            //实例化一个List对象进行赋值
            //用了反射中双A中的Activator来进行快速实例化List对象
            IList list = Activator.CreateInstance(fieldType) as IList;
            for(int i = 0; i < count; i++)
            {
                //得到List中泛型的类型
                list.Add(LoadValue(fieldType.GetGenericArguments()[0], keyName + i));
            }
            return list;
        }
        //读取Dictionary数据类型
        else if( typeof(IDictionary).IsAssignableFrom(fieldType))
        {
            //得到长度
            int count = PlayerPrefs.GetInt(keyName, 0);
            //实例化一个Dictionary对象进行赋值
            IDictionary dict = Activator.CreateInstance(fieldType) as IDictionary;
            Type[] kvType = fieldType.GetGenericArguments();
            for (int i = 0; i < count; i++)
            {
                dict.Add(LoadValue(kvType[0], keyName + "_key_" + i),
                        LoadValue(kvType[1], keyName + "_value_" + i));
            }
            return dict;
        }
        //读取自定义数据类型
        else
        {
            return LoadData(fieldType, keyName);
        }
    }
}
