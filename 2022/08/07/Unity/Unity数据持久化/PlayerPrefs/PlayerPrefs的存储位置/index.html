PlayerPrefs的存储位置

PlayerPrefs的存储位置

PlayerPrefs的存储位置

不同平台存储的位置不同

Windows平台

PlayerPrefs存储在HKCU\ Software\ [Company Name]\ [Product Name]项下的注册表中
其中Company Name和Product Name是在“Project Setting”中设置的名称

win+R运行 regedit
进入HKEY_CURRENT_USER分级下的SOFTWARE下的Unity下的UnityEditor
寻找Company Name和Product Name

Android平台

/data/data/包名/shared_prefs/pkg-name.xml

IOS平台

/Library/Preferences/[应用ID].plist

PlayerPrefs数据唯一性

PlayerPrefs中不同数据的唯一性是由key决定的，不同的key决定了不同的数据
同一项目中，如果不同数据key相同，会造成数据丢失
要保证数据不丢失就要建立一个保证key唯一的规则

题1

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class PlayerNO2
{
    public string name;
    public int age;
    public int atk;
    public int def;
    public List<Item> items;

    private string keyName;
    public void SaveData()
    {
        PlayerPrefs.SetString(keyName + "_玩家", name);
        PlayerPrefs.SetInt(keyName + "_年龄", age);
        PlayerPrefs.SetInt(keyName + "_攻击力", atk);
        PlayerPrefs.SetInt(keyName + "_防御力", def);
        PlayerPrefs.SetInt(keyName + "_道具数", items.Count);

        for (int i = 0; i < items.Count; i++)
        {
            PlayerPrefs.SetInt(keyName + "_道具ID" + i, items[i].id);
            PlayerPrefs.SetInt(keyName + "_道具" + i + "数量", items[i].num);
        }

        PlayerPrefs.Save();
    }

    public void LoadData(string keyName)
    {
        name = PlayerPrefs.GetString(keyName + "_玩家", "未命名");
        age = PlayerPrefs.GetInt(keyName + "_年龄", 17);
        atk = PlayerPrefs.GetInt(keyName + "_攻击力", 0);
        def = PlayerPrefs.GetInt(keyName + "_防御力", 0);

        int num = PlayerPrefs.GetInt(keyName + "_道具数", 0);
        items = new List<Item>();
        for (int i = 0; i < num; i++)
        {
            Item item = new Item();
            item.id = PlayerPrefs.GetInt(keyName + "_道具ID" + i);
            item.num = PlayerPrefs.GetInt(keyName + "_道具" + i + "数量");
            items.Add(item);
        }
    }
}
public class Lesson2 : MonoBehaviour
{
    
    // Start is called before the first frame update
    void Start()
    {
        PlayerNO2 p1 = new PlayerNO2();
        p1.LoadData("Player1");
        p1.SaveData();
        PlayerNO2 p2 = new PlayerNO2();
        p2.LoadData("Player2");
        p2.SaveData();
        print(p1.age);
        print(p2.age);
    }

    // Update is called once per frame
    void Update()
    {
        
    }
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 排行榜具体信息
/// </summary>
public class RankListInfo
{
    public List<RankInfo> rankList;

    public RankListInfo()
    {
        Load();
    }

    public void Add(string name, int score, int time)
    {
        rankList.Add(new RankInfo(name, score, time));
    }

    public void Save()
    {
        PlayerPrefs.SetInt("RankListNum", rankList.Count);
        for (int i = 0; i < rankList.Count; i++)
        {
            RankInfo rankInfo = rankList[i];
            PlayerPrefs.SetString("RankInfo_" + i, rankInfo.playerName);
            PlayerPrefs.SetInt("RankScore_" + i, rankInfo.playerScore);
            PlayerPrefs.SetInt("RankTime_" + i, rankInfo.playerTime);
        }
    }

    private void Load()
    {
        int rankNum = PlayerPrefs.GetInt("RankListNum", 0);
        rankList = new List<RankInfo>();
        for (int i = 0; i < rankNum; i++)
        {
            RankInfo rankInfo = new RankInfo(PlayerPrefs.GetString("RankInfo_" + i),
                                            PlayerPrefs.GetInt("RankScore_" + i),
                                            PlayerPrefs.GetInt("RankTime_" + i));
            rankList.Add(rankInfo);
        }
    }
}

/// <summary>
/// 排行榜单条信息
/// </summary>
public class RankInfo
{
    public string playerName;
    public int playerScore;
    public int playerTime;

    public RankInfo(string name, int score, int time)
    {
        playerName = name;
        playerScore = score;
        playerTime = time;
    }
}
public class RankList : MonoBehaviour
{
    // Start is called before the first frame update
    void Start()
    {
        RankListInfo rankListInfo = new RankListInfo();
        print(rankListInfo.rankList.Count);
        for (int i = 0; i < rankListInfo.rankList.Count; i++)
        {
            print("姓名:" + rankListInfo.rankList[i].playerName);
            print("成绩:" + rankListInfo.rankList[i].playerScore);
            print("时间:" + rankListInfo.rankList[i].playerTime);
        }
        rankListInfo.Add("xyz", 50, 100);
        rankListInfo.Save();
    }

    // Update is called once per frame
    void Update()
    {
        
    }
}
