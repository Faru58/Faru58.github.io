图片绘制和框绘制

图片绘制和框绘制

图片绘制

ScaleMode

  1. ScaleMode.StretchToFill：默认模式，始终填充满输入的Rect，会拉伸
  2. ScaleMode.ScaleAndCrop：通过宽高比计算图片，超出比例会裁剪
  3. ScaleMode.ScaleToFit：通过宽高比计算，自动调整图片，不会拉变形，保持图片完全显示的状态

alphaBlend：是否背景透明

imageAspect：自定义参与计算的宽高比（0代表图片默认的宽高比）

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Lesson5 : MonoBehaviour
{
    public Rect imgPos;
    public Texture tex;
    public ScaleMode mode = ScaleMode.ScaleToFit;
    public bool alpha = true;
    public float wh = 0;

    private void OnGUI()
    {
        //图片绘制(不会等比缩放)
        GUI.DrawTexture(imgPos, tex, mode, alpha, wh);
    }
}

框绘制

using System.Collections;

using UnityEngine;

public class Lesson5 : MonoBehaviour
{
    public Rect texPos;

    private void OnGUI()
    {
        //框绘制
        GUI.Box(texPos, "雀食蟀");
    }
}
